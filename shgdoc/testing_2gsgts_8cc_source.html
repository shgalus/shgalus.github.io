68 /*
69  * Returns autocovariance function of AR(1) with white noise variance
70  * sigma2 and phi1, ie. x(t) = phi1 * x(t - 1) + e(t), e(t) being
71  * white noise with variance sigma2.
72  *
73  * acfar1(0) - variance
74  * acfar1(1) = gamma(1)
75  * gamma(h) = acfar1(h), h = 0, 1, ..., n - 1.
76  *
77  * brockwell-davis-2006, page 81.
78  */
79 std::vector<double> acfar1(double sigma2, double phi1, size_t n) {
80  SHG_ASSERT(n > 0);
81  SHG_ASSERT(std::abs(phi1) < 1.0);
82  std::vector<double> g(n);
83  g[0] = sigma2 / (1.0 - phi1 * phi1);
84  for (std::vector<double>::size_type i = 1; i < n; i++)
85  g[i] = phi1 * g[i - 1];
86  return g;
87 }
