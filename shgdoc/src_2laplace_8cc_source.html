65 double weighted_median(const SHG::Vecdouble& x,
66  const SHG::Vecdouble& w) {
67  const size_t n = x.size();
68  if (n < 1 || w.size() != n)
69  throw invalid_argument(__func__);
70  double u = 0.0;
71  for (size_t i = 0; i < n; i++) {
72  if (w[i] < 0.0)
73  throw invalid_argument(__func__);
74  if (i && (x[i] < x[i - 1]))
75  throw invalid_argument(__func__);
76  u -= w[i];
77  }
78  if (u >= 0.0) // all weights equal to 0
79  throw invalid_argument(__func__);
80  int k = 0;
81  do {
82  u += 2.0 * w[k++];
83  } while (u < 0.0);
84  if (u > 0.0)
85  return x[k - 1];
86  int l = k;
87  do {
88  u += 2.0 * w[l++];
89  } while (u <= 0.0);
90  return 0.5 * (x[k - 1] + x[l - 1]);
91 }

111 double Unilapmixmod::estep() {
112  const double oldloglik = loglik;
113  double s;
114  int i, k;
115  loglik = 0.0;
116  for (i = 0; i < n; i++) {
117  const double xi = x(i);
118  s = 0.0;
119  for (k = 0; k < K; k++)
120  s += (psi(i, k) =
121  pi(k) *
122  Laplace_distribution(mu(k), lambda(k)).f(xi));
123  SHG_ASSERT(s > 0.0);
124  for (k = 0; k < K; k++)
125  psi(i, k) /= s;
126  loglik += log(s);
127  }
128  return loglik - oldloglik;
129 }

131 void Unilapmixmod::mstep() {
132  int i, k;
133  double ak, bk, muk;
134  SHG::Vecdouble w(n), w_sorted(n);
135  for (k = 0; k < K; k++) {
136  bk = 0.0;
137  for (i = 0; i < n; i++)
138  bk += (w(i) = psi(i, k));
139  pi(k) = bk / n;
140  for (i = 0; i < n; i++)
141  w_sorted(i) = w(order(i));
142  mu(k) = muk = weighted_median(x_sorted, w_sorted);
143  ak = 0.0;
144  for (i = 0; i < n; i++)
145  ak += w(i) * abs(x(i) - muk);
146  SHG_ASSERT(bk > 0.0);
147  if ((lambda(k) = ak / bk) <= 0.0)
148  throw Degenerate_distribution();
149  }

186 double Laplace_mixture::invcdf(double p) const {
187  if (p <= 0.0 || p >= 1.0)
188  throw Error();
189  double a = -1e3, b = 1e3, x = 0.0;
190  if (p <= cdf(a) || p >= cdf(b))
191  throw Error();
192 
193  // Bisection.
194  while (b - a > 1e-10)
195  if (cdf(x = 0.5 * (a + b)) < p)
196  a = x;
197  else
198  b = x;
199  return x;

236 void Laplace_mixture::moments() {
237  if (moments_calculated)
238  return;
239  double m1 = 0.0, m2 = 0.0, m3 = 0.0, m4 = 0.0;
240 
241  // Calculate the first four raw moments of the mixture.
242  for (size_t i = 0; i < n_; i++) {
243  const double wi = w[i];
244  const double mu1 = mu[i];
245  const double mu2 = sqr(mu1);
246  const double lambda2 = sqr(lambda[i]);
247  m1 += wi * mu1;
248  m2 += wi * (mu2 + 2.0 * lambda2);
249  m3 += wi * mu1 * (mu2 + 6.0 * lambda2);
250  m4 += wi * (mu2 * (mu2 + 12.0 * lambda2) +
251  24.0 * sqr(lambda2));
252  }
253  mean_ = m1;
254  const double sm = sqr(m1);
255  const double var = m2 - sm;
256  if (var <= 0.0)
257  throw Error();
258  sdev_ = sqrt(var);
259  skew_ = (m3 - 3.0 * m1 * m2 + 2.0 * m1 * sm) / sdev_ / var;
260  curt_ = (m4 - 4.0 * m1 * m3 + 6.0 * sm * m2 - 3.0 * sqr(sm)) /
261  var / var - 3.0;
262  moments_calculated = true;
263 }
