28 /**
29  * Calculates permutation entropy of a time series.
30  *
31  * Let \f$(x_i)_{i = 0}^{N - 1}\f$ be a time series and let \f$2 \leq
32  * L \leq N\f$ be a size of a sliding window. Let \f$\pi = (\pi_0,
33  * \pi_1, \ldots, \pi_{L - 1})\f$ be a permutation of the set \f$\{0,
34  * 1, \ldots, L - 1\}\f$. The window \f$(x_i)_{i = n}^{n + L - 1}\f$
35  * is of type \f$\pi\f$ if \f$x_{n + \pi_0} \prec x_{n + \pi_1} \prec
36  * \ldots \prec x_{n + \pi_{L - 1}}\f$, where \f$x_i \prec x_j\f$ if
37  * \f$x_i < x_j\f$ or \f$x_i = x_j\f$ and \f$i < j\f$. The relative
38  * frequency of \f$\pi\f$ in \f$(x_i)_{i = 0}^{N - 1}\f$ is then
39  * \f[\hat{p}(\pi) = \frac{1}{N - L + 1}|\{n \colon 0 \leq n \leq N -
40  * L, (x_i)_{i = n}^{n + L - 1} \mbox{is of type $\pi$} \}|\f] and the
41  * permutation entropy is \f[-\sum_{\pi} \hat{p}(\pi)\log_2
42  * \hat{p}(\pi).\f] Conf. \cite amigo-2010, page 29--30, and
43  * \cite bandt-pompe-2002, page 2.
44  *
45  * \tparam T linearly ordered set
46  * \param[in] x time series
47  * \param[in] L size of sliding window
48  * \returns permutation entropy
49  *
50  * \exception std::invalid_argument if \f$L < 2\f$ or \f$L >
51  * \mathrm{x.size()}\f$.
52  */
53 template <class T>
54 double permutation_entropy(const std::vector<T>& x, std::size_t L);

57 double permutation_entropy(const std::vector<T>& x,
58  const std::size_t L) {
59  if (L < 2 || L > x.size())
60  throw std::invalid_argument(__func__);
61  typedef std::vector<std::size_t> Permutation;
62  const std::size_t nl1 = x.size() - L + 1;
63  std::map<Permutation, std::size_t> c;
64  Permutation p(L), p0(L);
65  std::iota(p0.begin(), p0.end(), 0);
66  for (std::size_t n = 0; n < nl1; n++) {
67  p = p0;
68  std::stable_sort(p.begin(), p.end(),
69  [n, &x](std::size_t i, std::size_t j) {
70  return x[n + i] < x[n + j];
71  });
72  c[p]++;
73  }
74  const double NL1 = nl1;
75  double s = 0.0;
76  for (const auto& x : c) {
77  const double p = x.second / NL1;
78  s -= p * std::log(p);
79  }
80  return
81  1.4426950408889634073599246810018921374266 // 1 / ln 2
82  * s;
83 }
