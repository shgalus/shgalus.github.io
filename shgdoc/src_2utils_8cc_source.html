25 Vecchar wfread(const char* filename) {
26  using std::ifstream;
27  using std::ios;
28  ifstream f(filename, ios::in | ios::binary);
29  if (f && f.seekg(0, ios::end)) {
30  const ifstream::pos_type n = f.tellg();
31  // Here either f.fail() != false or n >= 0. See
32  // \cite ansi-cpp-2012, 27.7.2.3 (40), page 1035.
33  if (f && f.seekg(0, ios::beg)) {
34  // ifstream::pos_type may be long long (64 bit), size_t
35  // may be unsigned (32 bit), streamsize may be int (32
36  // bit)
37  if (n > std::numeric_limits<Vecchar::size_type>::max() ||
38  n > std::numeric_limits<std::streamsize>::max())
39  throw std::runtime_error(__func__);
40  SHG::Vecchar buf(static_cast<Vecchar::size_type>(n));
41  const std::streamsize ns =
42  static_cast<std::streamsize>(n);
43  if (f.read(buf.c_vec(), ns) && f.gcount() == ns)
44  return buf;
45  }
46  }
47  throw File_error(filename);
48 }

229 bool isdegenerate(const std::gslice& g) {
230  using std::size_t;
231  using std::valarray;
232  using std::inner_product;
233  using std::begin;
234  using std::end;
235 
236  const valarray<size_t>& l = g.size();
237  const valarray<size_t>& d = g.stride();
238  const size_t n = l.size();
239  if (n < 1 || n != d.size())
240  return true;
241  if (l.min() < 1 || d.min() < 1)
242  return true;
243  valarray<bool> addr(inner_product(begin(l), end(l), begin(d), 0)
244  - d.sum() + 1);
245  valarray<size_t> i(n);
246  for (;;) {
247  size_t kk = inner_product(begin(i), end(i), begin(d), 0);
248  SHG_ASSERT(kk < addr.size());
249  if (addr[kk])
250  return true;
251  addr[kk] = true;
252  for (size_t k = n - 1; ++i[k] >= l[k]; k--) {
253  if (k == 0)
254  return false;
255  i[k] = 0;
256  }
257  }
258 }
