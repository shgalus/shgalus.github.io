19 void ksone(T& cdf, Vecdouble& x, double& d, double& prob) {
20  const int n = x.size();
21  double dt, en = n, ff, fn, fo = 0.0;
22  std::sort(x.begin(), x.end());
23  d = 0.0;
24  for (int j = 0; j < n; j++) {
25  fn = (j + 1) / en;
26  ff = cdf(x[j]);
27  dt = SHG::max(std::abs(fo - ff), std::abs(fn - ff));
28  if (dt > d)
29  d = dt;
30  fo = fn;
31  }
32  en = std::sqrt(en);
33  prob = 1.0 - ksdist(d * en);
34  // With approximation (14.3.18) from
35  // \cite press-teukolsky-vetterling-flannery-2007, p. 737, the
36  // last line would be prob = 1.0 - ksdist(d * (en + 0.12 + 0.11 /
37  // en));
38 }

47 void ksoned(T& cdf, Vecint& x, double& d, double& prob) {
48  // For discrete distributions with jumps in integer numbers this
49  // eps is enough.
50  static const double eps = 1e-5;
51  const int n = x.size();
52  double dt, en = n, ff, fn, fe, fo = 0.0;
53  std::sort(x.begin(), x.end());
54  d = 0.0;
55  for (int j = 0; j < n;) {
56  while (++j < n && x[j] == x[j - 1]) {/* VOID */}
57  fn = j / en;
58  ff = cdf(x[j - 1]);
59  fe = cdf(x[j - 1] - eps);
60  dt = SHG::max(std::abs(fo - fe), std::abs(fn - ff));
