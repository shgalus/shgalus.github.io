185  static const int maxiter = 10000;
186  const int m = A.nrows();
187  const int n = A.ncols();
188  if (m <= 0 || n <= 0)
189  return 1; // invalid parameter m or n
190  {
191  const size_t ms = m;
192  const size_t ns = n;
193  if (p.size() != ns || C.size() != ns * (ns + 1) / 2 ||
194  A.nrows() != ms || A.ncols() != ns ||
195  b.size() != ms || x.size() != ns)
196  return 1; // invalid dimensions
197  }
198  const int g = m + n; // added row in simplex table
199  const int h = 2 * n + m; // added column in simplex table
200  const int // intervals of variable numbers
201  // x: 0 .. nv - 1
202  nv = n, // v: nv .. nu - 1
203  nu = 2 * n, // u: nu .. nz - 1
204  nz = 2 * n + m, // z: nz .. nw - 1
205  nw = 3 * n + m; // w: nw .. 3 * n + 2 * m - 1
206  Matdouble t; // simplex table
207  Vecint B, N; // basis and non-basis variables
208  int q = m; // last row to leave the basis
209  bool initiation = true; // true = initiation, false = recursion
210  int iter = 0;
211  int i, j, k, l = 0; // Dummy initialization for compiler.
212  double d, s;
213 
214  try {
215  t.resize(g + 1, h + 1);
216  B.resize(g);
217  N.resize(h);
218  }
219  catch (std::bad_alloc) {
220  return 2; // not enough memory
221  }
222 
223  // Initialize basis and non-basis variable numbers.
224  for (i = 0; i < g; i++)
225  B[i] = i < m ? nw + i : nz + (i - m);
226  for (i = 0; i < h; i++)
227  N[i] = i;
228 
229  // Initialize first m rows of the simplex table.
230  for (i = 0; i < m; i++) {
231  if (b[i] < 0.0) {
232  for (j = 0; j < n; j++)
233  t[i][j] = -A[i][j];
234  t[i][h] = -b[i];
235  } else {
236  for (j = 0; j < n; j++)
237  t[i][j] = A[i][j];
238  t[i][h] = b[i];
239  }
240  for (j = n; j < h; j++)

242  }
243 
244  // Initialize next n rows of the simplex table.
245  k = 0;
246  for (i = m; i < g; i++) {
247  t[i][i - m] = 2.0 * C[k++];
248  for (j = i - m + 1; j < n; j++)
249  t[i][j] = t[m + j][i - m] = 2.0 * C[k++];
250  for (j = n; j < 2 * n; j++)

252  t[i][n + (i - m)] = -1.0;
253  for (j = 2 * n; j < h; j++)
254  t[i][j] = A[j - 2 * n][i - m];
255  t[i][h] = -p[i - m];
256  if (t[i][h] < 0.0) {
257  for (j = 0; j <= h; j++)
258  t[i][j] = -t[i][j];
259  }
260  }
261 
262  // Initialize the g-th row of the simplex table.
263  for (j = 0; j < n; j++) {
264  s = 0.0;
265  for (i = 0; i < m; i++)
266  s += t[i][j];
267  t[g][j] = s;
268  }
269  for (j = n; j < h; j++)
270  t[g][j] = 0.0;
271  s = 0.0;
272  for (i = 0; i < m; i++)
273  s += t[i][h];
274  t[g][h] = s;
275 
276  for (;;) { // for initiation and recursion
277  if (++iter > maxiter)
278  return 3; // too many iteration steps
279 
280  // Find variable to enter the basis.
281  s = 0.0;
282  if (initiation) {
283  for (j = 0; j < h; j++)
284  // exclude u, v, z
285  if (N[j] < nv || N[j] >= nw)
286  if (t[g][j] > s)
287  s = t[g][l = j];
288  } else {
289  for (j = 0; j < h; j++)
290  if (N[j] < nw) {
291  // exclude w
292  if (N[j] >= nv && N[j] < nu) {
293  // side condition for v
294  k = N[j] - n;
295  for (i = 0; i < g && B[i] != k; i++) ;
296  if (i < g)
297  continue;
298  } else if (N[j] < nv) {
299  // side condition for x
300  k = N[j] + n;
301  for (i = 0; i < g && B[i] != k; i++) ;
302  if (i < g)
303  continue;
304  } else if (N[j] >= nu && N[j] < nz &&
305  t[g][j] < 0.0) {
306  // adjust u, as we have u1 and u2
307  for (i = 0; i <= g; i++)
308  t[i][j] = -t[i][j];
309  }
310  if (t[g][j] > s)
311  s = t[g][l = j];
312  }
313  }
314 
315  // If s < eps, there are three possibilities:
316  // - no feasible solution,
317  // - the method finishes successfully in recursion phase,
318  // - the phase is changed from initiation to recursion.
319 
320  if (s < eps) {
321  if (t[g][h] > eps)
322  // no feasible solution
323  return initiation ? 4 : 5;
324  if (!initiation)
325  break; // finished successfully
326 
327  // Change the phase.
328  initiation = false;
329  q = g;
330 
331  // Discard unused components of z.
332  for (i = m; i < g; i++)
333  if (t[i][h] < 0.0) {
334  for (j = 0; j <= h; j++)
335  t[i][j] = -t[i][j];
336  }
337 
338  // Calculate last row for the new objective function.
339  for (j = 0; j <= h; j++) {
340  s = 0.0;
341  for (i = 0; i < g; i++)
342  if (B[i] >= nz && B[i] < nw)
343  s += t[i][j];
344  if (j < h && N[j] >= nz && N[j] < nw)
345  s -= 1.0;
346  t[g][j] = s;
347  }
348  continue;
349  }
350 
351  // Find variable to leave the basis.
352  k = -1;
353  for (i = 0; i < q; i++)
354  // discard w in recursion
355  if ((initiation || B[i] < nw) && t[i][l] >= eps) {
356  d = t[i][h] / t[i][l];
357  if (k < 0 || d < s) {
358  k = i;
359  s = d;
360  }
361  }
362 
363  if (k < 0)
364  // unbounded solution
365  return initiation ? 6 : 7;
366 
367  // Change variables.
368  i = B[k];
369  B[k] = N[l];
370  N[l] = i;
371 
372  // Transform the simplex table around t[k][l].
373  s = t[k][l];
374  t[k][l] = 1.0;
375  for (j = 0; j <= h; j++)
376  t[k][j] /= s;
377  for (i = 0; i <= g; i++)
378  if (i != k) {
379  d = t[i][l];
380  t[i][l] = 0.0;
381  for (j = 0; j <= h; j++)
382  t[i][j] -= t[k][j] * d;
383  }
384  } // for initiation and recursion
385 
386  // Put the solution.
387  for (i = 0; i < n; i++)
388  x[i] = 0.0;
389 
390  for (i = 0; i < g; i++)
391  if (B[i] < n)
392  x[B[i]] = t[i][h];
393 
394  // Calculate minimum value.
395  f = 0.0;
396  k = 0;
397  for (i = 0; i < n; i++) {
398  s = 0.0;
399  d = C[k++];
400  for (j = i + 1; j < n; j++)
401  s += C[k++] * x[j];
402  f += x[i] * (2.0 * s + d * x[i] + p[i]);
403  }
404  return 0;
405 }
