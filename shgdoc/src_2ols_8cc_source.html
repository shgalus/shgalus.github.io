29 OLS::OLS(const Matdouble& X,
30  const Vecdouble& y,
31  bool add_intercept) :
32  problem_name_(),
33  n_(X.nrows()),
34  k_(add_intercept ? X.ncols() + 1: X.ncols()),
35  dof_(n_ - k_),
36  intercept_(add_intercept),
37  beta_(k_),
38  ser_(),
39  r2_(),
40  rbar2_(),
41  var_(),
42  stderr_(k_),
43  cov_(k_, k_),
44  fitted_(n_),
45  residuals_(n_),
46  rss_(0.0),
47  tss_(0.0),
48  ess_(),
49  mean_y_(),
50  stddev_y_(0.0),
51  positive_y_(true),
52  variation_y_(),
53  tstat_(k_, -1.0),
54  pvalt_(k_),
55  fstat_(-1.0),
56  pvalf_(),
57  dw_d_(-1.0),
58  dw_pvalpos_(-1.0),
59  dw_pvalneg_(-1.0) {
60  if (n_ < 1 || k_ < 1 || n_ < k_ ||
61  static_cast<std::size_t>(n_) != y.size())
62  throw Invalid_argument();
63 
64  estimate(X, y);
65 
66  // Check cov_. Is this really needed? At least for cholesky?
67  for (int i = 0; i < k_; i++)
68  if (!(cov_(i, i) > 0.0))
69  throw Singular_covariance_matrix();
70 
71  // Calculate mean of dependent variable, standard deviation of
72  // dependent variable, coefficient of variation of dependent
73  // variable and total sum of squares.
74 
75  mean_y_ = y(0);
76  for (int i = 1; i < n_; i++) {
77  const double yi = y(i);
78  const double d = yi - mean_y_;
79  mean_y_ += d / (i + 1);
80  stddev_y_ += (yi - mean_y_) * d;
81  if (yi <= 0.0)
82  positive_y_ = false;
83  }
84  if (!(stddev_y_ >= 0.0))
85  throw Internal_error();
86  if (add_intercept) {
87  tss_ = stddev_y_;
88  } else {
89  for (int t = 0; t < n_; t++)
90  tss_ += sqr(y(t));
91  }
92  stddev_y_ = std::sqrt(stddev_y_ / n_);
93  if (positive_y_)
94  variation_y_ = stddev_y_ / mean_y_;
95 
96  if (dof_ == 0) {
97  // If dof_ == 0, then n_ == k_. We assign:
98  fitted_ = y;
99  residuals_ = 0.0;
100  rss_ = 0.0;
101  ess_ = tss_;
102  r2_ = 1.0;
103  rbar2_ = 1.0;
104  var_ = 0.0;
105  ser_ = 0.0;
106  stderr_ = 0.0;
107  cov_ = 0.0;
108  return;
109  }
110 
111  if (dof_ <= 0 || n_ <= k_)
112  throw Internal_error();
113 
114  // Calculate fitted values, residuals, rss.
115  if (add_intercept) {
116  for (int t = 0; t < n_; t++) {
117  double s = beta_(0);
118  for (int i = 1; i < k_; i++)
119  s += beta_(i) * X(t, i - 1);
120  fitted_(t) = s;
121  const double e = y(t) - fitted_(t);
122  residuals_(t) = e;
123  rss_ += e * e;
124  }
125  } else {
126  for (int t = 0; t < n_; t++) {
127  double s = 0.0;
128  for (int i = 0; i < k_; i++)
129  s += beta_(i) * X(t, i);
130  fitted_(t) = s;
131  const double e = y(t) - fitted_(t);
132  residuals_(t) = e;
133  rss_ += e * e;
134  }
135  }
136 
137  ess_ = tss_ - rss_;
138  if (ess_ < 0.0) {
139  rss_ = tss_;
140  ess_ = 0.0;
141  }
142 
143  var_ = 1.0 / dof_ * rss_;
144  ser_ = std::sqrt(var_);
145 
146  // Calculate covariance matrix of parameters, standard errors of
147  // parameters and t statistics.
148  for (int i = 0; i < k_; i++) {
149  const double s = stderr_(i) = std::sqrt(cov_(i, i) *= var_);
150  const double b = beta_(i);
151  if (std::abs(b / std::numeric_limits<double>::max()) < s) {
152  tstat_(i) = std::abs(b) / s;
153  pvalt_(i) = 2.0 * (1.0 - probst(tstat_(i), dof_));
154  }
155  for (int j = i + 1; j < k_; j++)
156  cov_(j, i) = cov_(i, j) *= var_;
157  }
158 
159  // Calculate F-statistic.
160  const double eps = n_ / std::numeric_limits<double>::max();
161  double q;
162  if (tss_ > 0.0 && (q = rss_ / tss_) > eps) {
163  rbar2_ = 1.0 - static_cast<double>(n_ - 1) / dof_ * q;
164  r2_ = 1.0 - q;
165  if (add_intercept && k_ > 1)
166  // fstat_ = (ess_ / (k_ - 1)) / (rss_ / (n_ - k_));
167  fstat_ = (1.0 / q - 1.0) * (n_ - k_) / (k_ - 1);
168  else
169  // fstat_ = (ess_ / k_) / (rss_ / (n_ - k_));
170  fstat_ = (1.0 / q - 1.0) * (n_ - k_) / k_;
171  const int dfnum = add_intercept && k_ > 1 ? k_ - 1 : k_;
172  pvalf_ = 1.0 - cdffdist(dfnum, n_ - k_, fstat_);
173  } else {
174  // If tss_ == 0, then rss_ == ess_ == 0. All y(i) are
175  // constant. If there is no intercept, then y(i) = 0 for all
176  // i and a model with b(i) = 0 explains all. If there is an
177  // intercept, then y(i) = const for all i and a model with
178  // b(0) = const explains all.
179  r2_ = rbar2_ = 1.0;
180  }
181 }

183 void OLS::dw() {
184  if (!intercept_ || k_ <= 1 || n_ < 6 || dof_ < 4 ||
185  rss_ < tolerance_)
186  return;
187  double x = 0.0;
188  for (int t = 1; t < n_; t++)
189  x += sqr(residuals_[t] - residuals_[t - 1]);
190  x /= rss_;
191  if (x < 0.0)
192  x = 0.0;
193  else if (x > 4.0)
194  x = 4.0;
195  dw_d_ = x;
196  try {
197  dw_pvalpos_ = dwcdf(n_, k_ - 1, x, true, 0.00001, 15);
198  }
199  catch (std::range_error) {}
200  try {
201  dw_pvalneg_ = 1.0 - dwcdf(n_, k_ - 1, x, false, 0.00001, 15);
202  }
203  catch (std::range_error) {}
204 }

296 void OLS::estimate(const Matdouble& X, const Vecdouble& y) {
297  Vecdouble xty(k_);
298  if (intercept_) {
299  int i, i1, j, j1, l;
300  double s;
301  cov_(0, 0) = n_;
302  for (j = 1; j < k_; j++) {
303  s = 0.0;
304  j1 = j - 1;
305  for (l = 0; l < n_; l++)
306  s += X(l, j1);
307  cov_(0, j) = s;
308  }
309  s = 0.0;
310  for (l = 0; l < n_; l++)
311  s += y(l);
312  xty(0) = s;
313  for (i = 1; i < k_; i++) {
314  i1 = i - 1;
315  s = 0.0;
316  for (j = i; j < k_; j++) {
317  j1 = j - 1;
318  s = 0.0;
319  for (l = 0; l < n_; l++)
320  s += X(l, i1) * X(l, j1);
321  cov_(i, j) = s;
322  }
323  s = 0.0;
324  for (l = 0; l < n_; l++)
325  s += X(l, i1) * y(l);
326  xty(i) = s;
327  }
328  } else {
329  int i, j, l;
330  double s;
331  for (i = 0; i < k_; i++) {
332  for (j = i; j < k_; j++) {
333  s = 0.0;
334  for (l = 0; l < n_; l++)
335  s += X(l, i) * X(l, j);
336  cov_(i, j) = s;
337  }
338  s = 0.0;
339  for (l = 0; l < n_; l++)
340  s += X(l, i) * y(l);
341  xty(i) = s;
342  }
343  }
344  try {
345  cholesky(cov_, tolerance_);
346  } catch (std::range_error) {
347  throw Singular_covariance_matrix();
348  }
349 
350  for (std::size_t i = 0; i < cov_.nrows(); i++)
351  if (!std::isfinite(cov_(i, i)))
352  throw Singular_covariance_matrix();
353 
354  for (int i = 0; i < k_; i++) {
355  double s = 0.0;
356  for (int j = 0; j < k_; j++)
357  s += cov_(i, j) * xty(j);
358  beta_(i) = s;
359  }
360 }
