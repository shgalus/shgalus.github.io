255 Sample::Histdata Sample::histogram() const {
256  double h = 2.0 * pow(n, -1.0 / 3.0) * iqr();
257  /*
258  * It may happen that iqr() == 0. This means that cdf has a jump
259  * in a certain point x0 which is of the length at least 1/2. In
260  * this case at least half of the sample has a constant value. We
261  * are trying with the full range.
262  */
263  if (h < eps)
264  h = 2.0 * pow(n, -1.0 / 3.0) * range();
265  if (h < eps)
266  return Histdata(0.0, 0.0);
267  return histogram(iceil(range() / h));
268 }

318 double mle_lsd(const double mean) {
319  using std::log;
320  using std::numeric_limits;
321  SHG_VALIDATE(mean > 1.0);
322  double a = numeric_limits<double>::epsilon();
323  double b = 1.0 - a;
324  double p = 0.0; // shut up compiler warnings
325  auto fprim = [mean](const double p) -> double {
326  const double z = 1.0 - p;
327  return mean / p + 1.0 / log(z) / z;
328  };
329  if (fprim(a) < 0.0 || fprim(b) > 0.0)
330  throw runtime_error("sign error in mle_lsd");
331  // Bisection.
332  while (b - a > 1e-8)
333  if (fprim(p = 0.5 * (a + b)) > 0.0)
334  a = p;
335  else
336  b = p;
337  return p;
338 }

360 void mle_negative_binomial(const SHG::Vecint& x, double& t, double& p) {
361  const int n = x.size();
362  if (n < 1)
363  throw invalid_argument("n < 1 in mle_negative_binomial");
364  for (int i = 0; i < n; i++)
365  if (x(i) < 0)
366  throw invalid_argument(
367  "x(i) < 0 in mle_negative_binomial");
368  double mean, var;
369  {
370  Vecdouble z(n);
371  for (int i = 0; i < n; i++)
372  z(i) = x(i);
373  mean_var(z, mean, var);
374  }
375  if (mean < 1e-13 || var < 1e-13)
376  throw runtime_error(
377  "degenerate sample in mle_negative_binomial");
378  // Find estimators of p and t by the method of moments.
379  p = mean / var;
380  if (p <= 0.0 || p >= 1.0)
381  throw runtime_error(
382  "invalid estimator of p in mle_negative_binomial");
383  t = mean * p / (1.0 - p);
384  auto f = [x, n, mean](const double t) -> double {
385  double s = 0.0;
386  for (int i = 0; i < n; i++) {
387  const double xi = x(i);
388  for (int j = 0; j < xi; j++)
389  s += 1.0 / (t + j);
390  }
391  s += n * std::log(t / (t + mean));
392  return s;
393  };
394  t = brent(f, 0.5 * t, 2.0 * t, 1e-5);
395  if (t <= 0.0)
396  throw runtime_error(
397  "invalid estimator of t in mle_negative_binomial");
398  p = t / (t + mean);
399 }

432 void Assessment_of_discrete_distribution::run() {
433  // geometric
434  {
435  const double p = 1.0 / mean_; // ML estimator of p
436  if (0.0 < p && p < 1.0) {
437  const double q = 1.0 - p;
438  auto cdf = [q](const double x) -> double {
439  if (x < 1.0)
440  return 0.0;
441  return 1.0 - std::pow(q, ifloor(x));
442  };
443  double d;
444  ksonedc(cdf, x, d, geometric_);
445  } else {
446  geometric_ = 0.0;
447  }
448  }
449  // poisson
450  {
451  const double
452  lambda = 1.0 / mean_; // ML estimator of lambda
453  SHG_ASSERT(lambda > 0.0);
454  auto cdf = [lambda](const double x) -> double {
455  return x < 0.0 ?
456  0.0 : 1.0 - gammad(lambda, ifloor(x) + 1);
457  };
458  double d;
459  ksonedc(cdf, x, d, poisson_);
460  }
461  // logarithmic
462  {
463  if (mean_ > 1.0) {
464  const double p = mle_lsd(mean_);
465  auto cdf = [p](const double x) -> double {
466  return cdf_lsd(x, p);
467  };
468  double d;
469  ksonedc(cdf, x, d, logarithmic_);
470  } else {
471  logarithmic_ = 0.0;
472  }
473  }
474  // negative binomial
475  {
476  double t, p;
477  double d;
478  try {
479  mle_negative_binomial(x, t, p);
480  auto cdf = [t, p](const double x) -> double {
481  return cdf_negative_binomial(x, t, p);
482  };
483  ksonedc(cdf, x, d, negbin_);
484  } catch (runtime_error) {
485  // we assume that if p or t cannot be easy estimated,
486  // the distribution is far from negative binomial
487  negbin_ = 0;
488  }
489  }
