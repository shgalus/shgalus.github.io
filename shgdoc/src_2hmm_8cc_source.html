63 int Normal_hmm::forwardbackward() {
64  size_t i, j, t, t1;
65  double sum, u;
66 
67  u = 0.0;
68  for (j = 0; j < s_; j++)
69  u += alpha(0, j) = p(j) * phi(0, j);
70  if (u <= 0.0)
71  return 1;
72  SHG_ASSERT(u > 0.0);
73  c(0) = u = 1.0 / u;
74  for (j = 0; j < s_; j++)
75  alpha(0, j) *= u;
76  logL = -log(u);
77  for (t = 1; t < T_; t++) {
78  t1 = t - 1;
79  u = 0.0;
80  for (j = 0; j < s_; j++) {
81  sum = 0.0;
82  for (i = 0; i < s_; i++)
83  sum += alpha(t1, i) * P(i, j);
84  u += alpha(t, j) = sum * phi(t, j);
85  }
86  // This may happen that in an iteration phi(t, j) is extremaly
87  // small but positive (like 1e-320). Thus alpha(t, j) may be
88  // set to an extremely big number which produces overflow in
89  // the next iteration.
90  if (u < 1e-70)
91  return 2;
92  SHG_ASSERT(u > 0.0);
93  c(t) = u = 1.0 / u;
94  logL -= log(u);
95  for (j = 0; j < s_; j++)
96  alpha(t, j) *= u;
97  }
98 
99  d(T_ - 1) = u = 1.0 / s_;
100  for (i = 0; i < s_; i++)
101  beta(T_ - 1, i) = u;
102  for (t1 = T_ - 1; t1 > 0; t1--) {
103  t = t1 - 1;
104  u = 0.0;
105  for (i = 0; i < s_; i++) {
106  sum = 0.0;
107  for (j = 0; j < s_; j++)
108  sum += P(i, j) * phi(t1, j) * beta(t1, j);
109  u += beta(t, i) = sum;
110  }
111  if (u <= 0.0)
112  return 3;
113  SHG_ASSERT(u > 0.0);
114  d(t) = u = 1.0 / u;
115  for (i = 0; i < s_; i++)
116  beta(t, i) *= u;
117  }
118 
119  // Calculate gamma(t, i) and aux(t).
120  for (t = 0; t < T_; t++) {
121  sum = 0.0;
122  for (i = 0; i < s_; i++) {
123  u = alpha(t, i) * beta(t, i);
124  sum += u;
125  gamma(t, i) = u;
126  }
127  if (sum <= 0.0)
128  return 4;
129  SHG_ASSERT(sum > 0.0);
130  for (i = 0; i < s_; i++)
131  gamma(t, i) /= sum;
132  aux(t) = sum;
133  }
134  return 0;
135 }

137 int Normal_hmm::baumwelch() {
138  const size_t T1 = T_ - 1;
139  Vecdouble sg(s_), sg1(s_);
140  double u, pij, sg1i, mui;
141  size_t i, j, t;
142 
143  for (i = 0; i < s_; i++) {
144  u = 0.0;
145  for (t = 0; t < T1; t++)
146  u += gamma(t, i);
147  sg1(i) = u;
148  if (u <= 0.0)
149  return 5;
150  SHG_ASSERT(u > 0.0);
151  sg(i) = u + gamma(T1, i);
152  SHG_ASSERT(sg(i) > 0.0);
153  }
154  for (i = 0; i < s_; i++)
155  p(i) = gamma(0, i);
156  for (i = 0; i < s_; i++) {
157  sg1i = sg1(i);
158  for (j = 0; j < s_; j++) {
159  pij = P(i, j);
160  u = 0.0;
161  for (t = 0; t < T1; t++)
162  u += d(t) * alpha(t, i) * beta(t + 1, j) *
163  pij * phi(t + 1, j) / aux(t);
164  P(i, j) = u / sg1i;
165  SHG_ASSERT(P(i, j) >= 0.0 && P(i, j) - 1.0 < 1e-8);
166  }
167  }
168  for (i = 0; i < s_; i++) {
169  u = 0.0;
170  for (t = 0; t < T_; t++)
171  u += gamma(t, i) * y(t);
172  mu(i) = u / sg(i);
173  }
174  for (i = 0; i < s_; i++) {
175  u = 0.0;
176  mui = mu(i);
177  for (t = 0; t < T_; t++)
178  u += gamma(t, i) * sqr(y(t) - mui);
179  SHG_ASSERT(u >= 0);
180  sigma(i) = u = sqrt(u / sg(i));
181  if (u <= 0.0)
182  return 6;
183  SHG_ASSERT(u > 0.0);
184  }
185  return 0;
186 }

188 void Normal_hmm::viterbi() {
189  static const double mind = -numeric_limits<double>::max();
190  const size_t T1 = T_ - 1;
191  Matdouble delta(T_, s_);
192  Matint psi(T_, s_);
193  Matdouble logp(s_, s_);
194  size_t i, j, t, t1, imax;
195  double d, dmax;
196 
197  for (i = 0; i < s_; i++)
198  for (j = 0; j < s_; j++)
199  logp(i, j) = P(i, j) > 0.0 ? log(P(i, j)) : mind;
200  for (i = 0; i < s_; i++) {
201  delta(0, i) = p(i) > 0.0
202  ? log(p(i)) + log(phi(0, i))
203  : mind;
204  psi(0, i) = 0; // unneeded
205  }
206  for (t = 1; t < T_; t++) {
207  t1 = t - 1;
208  for (j = 0; j < s_; j++) {
209  imax = 0;
210  dmax = delta(t1, 0) > mind && logp(0, j) > mind
211  ? delta(t1, 0) + logp(0, j)
212  : mind;
213  for (i = 1; i < s_; i++) {
214  d = delta(t1, i) > mind && logp(i, j) > mind
215  ? delta(t1, i) + logp(i, j)
216  : mind;
217  if (d > dmax) {
218  dmax = d;
219  imax = i;
220  }
221  }
222  delta(t, j) = dmax + log(phi(t, j));
223  psi(t, j) = imax;
224  }
225  }
226  imax = 0;
227  logprob = delta(T1, 0);
228  for (i = 1; i < s_; i++)
229  if (delta(T1, i) > logprob) {
230  logprob = delta(T1, i);
231  imax = i;
232  }
233  X(T1) = imax;
234  for (t = T1; t > 0; t--)
235  X(t - 1) = psi(t, X(t));
236 }

280  const Vecdouble& p,
281  const Vecdouble& mu,
282  const Vecdouble& sigma,
283  const std::size_t T,
284  Vecdouble& y,

289  s == sigma.size());
290  SHG_ASSERT(T > 0);
291 
292  Vecdouble q(s);
293  y.resize(T);
294  X.resize(T);
295  int i;
296  X[0] = i = g.finite(p);
297  y[0] = mu[i] + g.normal() * sigma[i];
298  for (size_t t = 1; t < T; t++) {
299  // Catastrophe.

304  }

307 void corr_hmm(const Vecdouble& delta,
308  const Matdouble& gamma,
309  const Vecdouble& G,
310  const double E,
311  const double v,
312  const int K,
313  Vecdouble& r) {
314  SHG_ASSERT(K > 0);
315  SHG_ASSERT(v > 0.0);
316  const int m = delta.size();
317  const double E2 = E * E;
318  r.resize(K + 1);
319  r(0) = 1.0;
320  Matdouble gammak = diagonal_matrix<double>(static_cast<size_t>(m));
321  double s, s1;
322  int i, j, k;
323  for (k = 1; k <= K; k++) {
324  // Calculate E(g(X_t) * g(X_{t + k})).
325  right_multiply_and_assign(gammak, gamma);
326  s = 0.0;
327  for (i = 0; i < m; i++) {
328  s1 = 0.0;
329  for (j = 0; j < m; j++)
330  s1 += G(j) * gammak(i, j);
331  s += delta(i) * G(i) * s1;
332  }
333  r(k) = (s - E2) / v;
334  }
335 }

339 void corr_nhmm_id(const Vecdouble& delta,
340  const Matdouble& gamma,
341  const Vecdouble& mu,
342  const Vecdouble& sigma,
343  const int K,
344  Vecdouble& r) {
345  double E = 0.0; // E(X_t)
346  double v = 0.0; // var(X_t)
347 
348  for (size_t i = 0; i < delta.size(); i++) {
349  E += delta(i) * mu(i);
350  v += delta(i) * (sqr(mu(i)) + sqr(sigma(i)));
351  }
352  v -= sqr(E);
353  // E(g(X_t) | C_t = i) = mu(i), so we have G = mu.
354  corr_hmm(delta, gamma, mu, E, v, K, r);
355 }

357 void corr_nhmm_abs(const Vecdouble& delta,
358  const Matdouble& gamma,
359  const Vecdouble& mu,
360  const Vecdouble& sigma,
361  const int K,
362  Vecdouble& r) {
363  static const double // sqrt(2/pi)
364  C = 2.0 * SHG::Constants::isqrt2pi<double>();
365  const int m = delta.size();
366  double E = 0.0; // E(X_t)
367  double v = 0.0; // var(X_t)
368  Vecdouble G(m); // E(g(X_t) | C_t = i)
369  double mi, si, z;
370 
371  for (int i = 0; i < m; i++) {
372  mi = mu(i);
373  si = sigma(i);
374  SHG_ASSERT(si > 0.0);
375  z = mi / si;
376  G(i) = C * si * exp(-0.5 * sqr(z)) +
377  mi * (1.0 - 2.0 * normal_integral(-z));
378  E += delta(i) * G(i);
379  v += delta(i) * (sqr(mi) + sqr(si));
380  }
381  v -= sqr(E);
382  corr_hmm(delta, gamma, G, E, v, K, r);
383 }

385 void corr_nhmm_sqr(const Vecdouble& delta,
386  const Matdouble& gamma,
387  const Vecdouble& mu,
388  const Vecdouble& sigma,
389  const int K,

392  double E = 0.0; // E(X_t)
393  double v = 0.0; // var(X_t)
394  Vecdouble G(m); // E(g(X_t) | C_t = i)
395  double m2, s2, z;
396 
397  for (int i = 0; i < m; i++) {
398  m2 = sqr(mu(i));
399  s2 = sqr(sigma(i));
400  G(i) = m2 + s2;
401  E += delta(i) * G(i);
402  z = m2 * (m2 + 6.0 * s2) + 3.0 * sqr(s2);
403  v += delta(i) * z;
404  }
405  v -= sqr(E);
406  corr_hmm(delta, gamma, G, E, v, K, r);
407 }
