31 /**
32  * Generating stationary Gaussian time series given its autocovariance
33  * function. The implementation follows that of \cite davies-2001 (<a
34  * href = "StatGaus.pdf">local copy</a>).
35  *
36  * The elements \f$g_k\f$ in (2) of \cite davies-2001 can be
37  * calculated as \f[ g_k = c_0 + 2 \sum_{j = 1}^{n - 1} c_j \cos
38  * \frac{\pi j k}{n} + (-1)^k c_n, \quad k = 0, \ldots, n. \f]
39  *
40  * The elements \f$X_j\f$ in (3) of \cite davies-2001 can be
41  * calculated as \f[ X_j = \frac{1}{2 \sqrt{n}} \left[ u_0 \sqrt{g_0}
42  * + 2 \sum_{k = 1}^{n - 1} \sqrt{g_k} \left( u_k \cos \frac{2 \pi j
43  * k}{2n} - v_k \sin \frac{2 \pi j k}{2n} \right) + (-1)^j u_n
44  * \sqrt{g_n} \right], \quad j = 0, \ldots, n, \f] where \f$u_k = \Re
45  * z_k\f$, \f$v_k = \Im z_k\f$ for all \f$k\f$.
46  *
47  * %Sample implementations of cosine transform and real transform
48  * using <a href = "http://www.fftw.org/">FFTW library</a> are
49  * presented in cosft_fftw() and realft_fftw().
50  *
51  * Implementations using
52  * \cite press-teukolsky-vetterling-flannery-2007 could be as follows:
53  *
54  * \code
55  * #define _USESTDVECTOR_
56  * #include "nr3.h"
57  * #include "fourier.h"
58  *
59  * void cosft_numrec(const std::vector<double>& h,
60  * std::vector<double>& H) {
61  * H = h;
62  * cosft1(H);
63  * }
64  *
65  * void realft_numrec(const std::vector<std::complex<double>>& z,
66  * std::vector<double>& X) {
67  * const std::size_t n = z.size() - 1;
68  * const double c = 1.0 / (2.0 * std::sqrt(n));
69  * std::vector<std::complex<double>> data(2 * z.size() - 2);
70  * for (std::size_t i = 0; i < z.size(); i++)
71  * data[i] = z[i];
72  * for (std::size_t i = z.size(); i < data.size(); i++)
73  * data[i] = std::conj(data[data.size() - i]);
74  * four1(data, 1);
75  * for (std::size_t i = 0; i < z.size(); i++)
76  * X[i] = c * data[i].real();
77  * }
78  * \endcode
79  *
80  * \note Given the sequence (1) of \cite davies-2001 we calculate
81  * \f$g_k\f$ and \f$Z_k\f$ only for \f$k = 0, \ldots, n\f$ because
82  * \f$g_k = g_{2n - k}\f$, \f$Z_k = \overline{Z}_{2n - k}\f$ for \f$n
83  * < k < 2n\f$.
84  */
