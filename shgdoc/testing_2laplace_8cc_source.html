129 void test_laplace_mixture() {
130  Vecdouble w(3), mu(3), lambda(3);
131  w(0) = 1.0 / 2.0; mu(0) = -1.0; lambda(0) = 1.0;
132  w(1) = 1.0 / 3.0; mu(1) = 0.0; lambda(1) = 2.0;
133  w(2) = 1.0 / 6.0; mu(2) = 2.0; lambda(2) = 1.0;
134 
135  // Moments about zero of this mixture:
136  const double m1 = -1.0 / 6.0;
137  // const double m2 = 31.0 / 6.0;
138  // const double m3 = -1.0 / 6.0;
139  // const double m4 = 967.0 / 6.0;
140  // Central moments of this mixture:
141  const double nu2 = 185.0 / 36.0; // variance
142  const double nu3 = 520.0 / 216.0;
143  const double nu4 = 209841.0 / 1296.0;
144 
145  const double sdev = sqrt(nu2);
146  const double skew = nu3 / nu2 / sdev;
147  const double curt = nu4 / sqr(nu2) - 3.0;
148 
149  Laplace_mixture m(w, mu, lambda);
150  const double accu = 1e-14;
151  SHG_ASSERT(abs(m1 - m.mean()) < accu);
152  SHG_ASSERT(abs(sdev - m.sdev()) < accu);
153  SHG_ASSERT(abs(skew - m.skew()) < accu);
154  SHG_ASSERT(abs(curt - m.curt()) < accu);
155 
156  // Test generate().
157  Vecdouble x;
158  SHG::MZT mzt;
159  m.generate(mzt, 4000, x);
160 
161  // Test EM algorithm.
162  Unilapmixmod u(x, 3);
163  u.pi = w;
164  u.mu = mu;
165  u.lambda = lambda;
166  u.estep();
167  u.mstep();
168  int iter = 0;
169  for (;;) {
170  iter++;

172  break;
173  u.mstep();
174  }
175  // Results should be:

177  w0(0) = 0.46334203;
