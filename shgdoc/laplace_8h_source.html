32 /**
33  * Laplace distribution.
34  *
35  * The Laplace distribution is a probability distribution on
36  * \f$\mathbf{R}\f$ given by the density function \f[ f(x) =
37  * \frac{1}{2 \lambda} \exp \left( -\frac{|x - \mu|}{\lambda} \right),
38  * \quad \mu \in \mathbf{R}, \quad \lambda > 0 \f] (\cite fisz-1969,
39  * p. 170). Its probability distribution function is \f[ F(x) =
40  * \left\{ \begin{array}{ll} \frac{1}{2} \exp \left( \frac{x -
41  * \mu}{\lambda} \right) & \mbox{for $x \leq \mu$,} \\ 1 - \frac{1}{2}
42  * \exp \left( -\frac{x - \mu}{\lambda} \right) & \mbox{for $x \geq
43  * \mu$.} \end{array} \right. \f] The inverse probability distribution
44  * function is \f[ F^{-1}(p) = \left\{ \begin{array}{ll} \mu + \lambda
45  * \ln(2p) & \mbox{for $0 < p \leq 0.5$,} \\ \mu - \lambda \ln(2 - 2p)
46  * & \mbox{for $0.5 \leq p < 1$.} \end{array} \right. \f]
47  */
48 class Laplace_distribution {

50  /**
51  * Constructs Laplace distribution. \exception
52  * std::invalid_argument if \c lambda is not positive \warning
53  * lambda is not the variance!
54  */
55  Laplace_distribution(double mu = 0, double lambda = 1);

70  */
71  double invcdf(double p);

93 /**
94  * Weighted median.
95  *
96  * Let \f$f \colon \mathbf{R} \rightarrow \mathbf{R}\f$, \f$ f(\mu) =
97  * \sum_{i = 1}^n w_i|x_i - \mu|\f$, where \f$w_i \geq 0\f$, \f$x_1
98  * \leq x_2 \leq \ldots \leq x_n\f$ and not all \f$w_i\f$ are equal to
99  * 0. The function \f$f\f$ reaches its minimum either in a single
100  * point or on an interval. The function \c weighted_median returns
101  * this point if it is unique. If it is not unique, the function
102  * returns the middle point of the interval on which the minimum is
103  * reached.
104  *
105  * \exception std::invalid_argument if not <tt>a.size() == b.size() >
106  * 0</tt> or if not <tt>a[i] >= 0</tt> for all \c i or if sum of
107  * <tt>a[i]</tt> is not greater than 0 or if <tt>b[i]</tt> are not
108  * sorted non-decreasingly.
109  *
110  * \warning Due to rounding errors, the function may incorrect
111  * recognize unique and non-unique minima.
112  *
113  * \implementation Let us denote \f$x_0 = -\infty\f$, \f$x_{n + 1} =
114  * +\infty\f$. On each interval \f$x_{j - 1} \leq \mu \leq x_j\f$,
115  * \f$j = 1, \ldots, n + 1\f$, the function \f$f\f$ may be written as
116  * \f[ f(\mu) = \sum_{i = 1}^{j - 1} w_i(\mu - x_i) + \sum_{i = j}^n
117  * w_i(x_i - \mu) = u_j \mu + v_j, \f] where \f[ u_j = \sum_{i = 1}^{j
118  * - 1} w_i - \sum_{i = j}^n w_i. \f] We have \f$u_1 = -\sum_{i = 1}^n
119  * w_i < 0\f$, \f$u_{n + 1} = \sum_{i = 1}^n w_i > 0\f$ and \f$u_{j +
120  * 1} - u_j = 2w_j \geq 0\f$ for \f$j = 1, \ldots, n\f$, so the
121  * sequence \f$u_1, \ldots, u_{n + 1}\f$ is non-decreasing. Let \f[k =
122  * \min \left\{ j \in \{1, \ldots, n + 1 \} \colon u_j \geq 0
123  * \right\},\f] \f[l = \min \left\{ j \in \{1, \ldots, n + 1 \} \colon
124  * u_j > 0 \right\}.\f] For \f$j < k\f$ we have \f$u_j < 0\f$, so
125  * \f$f\f$ is strictly decreasing on \f$(-\infty, x_{k - 1}]\f$. For
126  * \f$j \geq l\f$ we have \f$u_j > 0 \f$, so \f$f\f$ is strictly
127  * increasing on \f$[x_{l - 1}, +\infty)\f$. On \f$[x_{k - 1}, x_{l -
128  * 1}]\f$ the function \f$f\f$ is constant. If this interval reduces
129  * to a point, ie. \f$k = l\f$, the minimum is uniquely reached at
130  * \f$x_{k - 1}\f$.
131  */
132 double weighted_median(const SHG::Vecdouble& x,
133  const SHG::Vecdouble& w);
