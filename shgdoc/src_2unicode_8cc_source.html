99 /**
100  * Decodes UTF-8 sequence contained in s starting from position i. If
101  * the sequence is correct, its Unicode code is returned and the
102  * position i is set to the beginning of the next sequence. If the
103  * sequence is incorrect, std::invalid_argument is thrown and the
104  * position in i remains undefinite.
105  */
106 std::uint32_t getc(const std::string& s, std::string::size_type& i) {
107  std::uint32_t u;
108  int nbytes;
109  unsigned char b = s[i];
110  if (b < 0x80u) {
111  u = b;
112  nbytes = 0;
113  } else if (b < 0xc0u) {
114  goto error;
115  } else if (b < 0xe0u) {
116  u = b & 0x1fu;
117  nbytes = 1;
118  } else if (b < 0xf0u) {
119  u = b & 0x0fu;
120  nbytes = 2;
121  } else if (b < 0xf8u) {
122  u = b & 0x07u;
123  nbytes = 3;
124  } else if (b < 0xfcu) {
125  u = b & 0x03u;
126  nbytes = 4;
127  } else if (b < 0xfeu) {
128  u = b & 0x01u;
129  nbytes = 5;
130  } else {
131  goto error;
132  }
133  for (int j = 0; j < nbytes; j++)
134  if (++i >= s.size() ||
135  ((b = s[i]) & 0xc0u) != 0x80u)
136  goto error;
137  else
138  (u <<= 6) += b & 0x3fu;
139  /* Check for overlong sequence. */
140  if (u < 0x80u) {
141  if (nbytes > 0)
142  goto error;
143  } else if (u < 0x0800u) {
144  if (nbytes > 1)
145  goto error;
146  } else if (u < 0x10000u) {
147  if (nbytes > 2)
148  goto error;
149  } else if (u < 0x200000u) {
150  if (nbytes > 3)
151  goto error;
152  } else if (u < 0x4000000u) {
153  if (nbytes > 4)
154  goto error;
155  } else if (u < 0x80000000u) {
156  if (nbytes > 5)
157  goto error;
158  }
159  /* Check for UTF-16 surrogate. */
160  if (u >= 0xd800u && u < 0xe000u)
161  goto error;
162  return u;
163 error:
164  throw std::invalid_argument(__func__);
165 }
