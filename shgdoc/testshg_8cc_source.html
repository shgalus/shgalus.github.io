103 void Test_shg::run(int argc, char* argv[]) {
104  using std::cout;
105  using std::fixed;
106  using std::flush;
107  using std::ofstream;
108  using std::setprecision;
109  using std::setw;
110  typedef std::vector<Module>::size_type vmst;
111  if (argc <= 1) {
112  for (vmst i = 0; i < mm.size(); i++)
113  mm[i].do_test = true;
114  } else {
115  for (int i = 1; i < argc; i++) {
116  const char* s = argv[i];

121  SHG_ASSERT(it != mm.end());
122  SHG_ASSERT(!it->do_test);
123  it->do_test = true;
124  }
125  }
126  std::string::size_type maxlen = 0;
127  for (vmst i = 0; i < mm.size(); i++)
128  if (mm[i].name.size() > maxlen)
129  maxlen = mm[i].name.size();
130  unsigned total = 0, passed = 0;
131  double elapsed, total_elapsed = 0.0;
132  SHG::Timer t;
133  logfile.exceptions(ofstream::failbit | ofstream::badbit);
134  logfile.open("testshg.log");
135  const std::ios_base::fmtflags flags = logfile.flags();
136  {
137  std::time_t rawtime;
138  std::time(&rawtime);
139  cout << "Testing SHG on "
140  << std::asctime(std::localtime(&rawtime)) << flush;
141  }
142  for (vmst i = 0; i < mm.size(); i++)
143  if (mm[i].do_test) {
144  total++;
145  const std::string& s = mm[i].name;
146  cout << "Testing " << s << ' '

148  << ' ' << flush;
149  logfile.flags(flags);

153  elapsed = t.elapsed();
154  total_elapsed += elapsed;
155  passed++;
156  cout << "passed. ("

158  << setw(6) << elapsed << " seconds)\n";
159  }
160  catch (const SHG::Exception& e) {
161  cout << "failed.\n";
162  e.print(prgname, cout);
163  }
164  catch (const std::exception& e) {
165  cout << "failed.\n";
166  SHG::error(e.what(), prgname, cout);
167  }
168  }
169  logfile.close();
170  cout << "Total number of tests: " << setw(3) << total
171  << ". (" << fixed << setprecision(3) << setw(6)
172  << total_elapsed << " seconds)\n"
173  << "Number of tests passed: " << setw(3) << passed << ".\n"
174  << "Number of tests failed: " << setw(3)
175  << total - passed << ".\n"

177 }
