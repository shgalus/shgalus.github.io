40 Date::Date(const char *s) : abs() {
41  /* The longest name of the month in polish_month[] and
42  roman_month[] is 13 characters long. */
43  static const int lnom = 13;
44 
45  if (const char* const sep = strchr(s, ' ')) {
46  /* "22 stycznia 1901" or "22 I 1901" */
47  int m, d;
48  switch (sep - s) {
49  case 1:
50  d = a1toi(s);
51  break;
52  case 2:
53  if (s[0] == '0')
54  throw Bad_date();
55  d = a2toi(s);
56  break;
57  default:
58  throw Bad_date();
59  }
60  const char* sep1 = strchr(sep + 1, ' ');
61  if (sep1 == 0 || sep1 == sep + 1)
62  throw Bad_date();
63  const int len = sep1 - (sep + 1);
64  if (len > lnom)
65  throw Bad_date();
66  char mnth[lnom + 1];
67  strncpy(mnth, sep + 1, len)[len] = 0;
68  for (m = 1; m <= 12; m++)
69  if (strcmp(mnth, polish_month[m]) == 0 ||
70  strcmp(mnth, roman_month[m]) == 0)
71  break;
72  if (m > 12 || strlen(++sep1) != 4)
73  throw Bad_date();
74  *this = Date(d, m, a4toi(sep1));
75  } else if (strchr(s, '.')) {
76  /* "22.01.1901" */
77  if (strlen(s) != 10)
78  throw Bad_date();
79  if (s[2] != '.' || s[5] != '.')
80  throw Bad_date();
81  *this = Date(a2toi(s), a2toi(s + 3), a4toi(s + 6));
82  } else {
83  /* "19010122" */
84  if (strlen(s) != 8)
85  throw Bad_date();
86  *this = Date(a2toi(s + 6), a2toi(s + 4), a4toi(s));
87  }
88 }

90 Date::Date(int n, int d, int m, int y) : abs() {
91  if (n == 0 || n > 5 || n < -5 || d < sun || d > sat ||
92  !correct(1, m, y))
93  throw Bad_date();
94  if (n > 0) {
95  const Date d1(1, m, y);
96  const int dow = d1.dayofweek();
97  const int z = d < dow ? d + (7 - dow) : d - dow;
98  *this = Date(d1 + z + (n - 1) * 7);
99  } else {
100  const bool leap = Date::leap(y);
101  const int last = daytab[leap][m];
102  const Date d1(last, m, y);
103  const int dow = d1.dayofweek();
104  const int z = d > dow ? d - (7 + dow) : d - dow;
105  *this = Date(d1 + z + (n + 1) * 7);
106  }
107  if (month() != m)
108  throw Bad_date();
109 }

130  int32 x = abs;

132  if (x == 1) {
133  y = 1;

233 Date Date::easter(int y) {
234  int p = y / 100;
235  int q = 4 + p - p / 4;
236  int r = q % 7;
237  q = 15 + p - p / 4 - (8 * p + 13) / 25;
238  int s = (19 * (y % 19) + q % 30) % 30;
239  if (s > 27 && (s == 29 || y % 19 > 10))
240  --s;
241  q = (2 * (y % 4) + 4 * (y % 7) - s + r) % 7;
242  while (q < 0)
243  q += 7;
244  return (s += q) > 9 ?
245  Date(s - 9, 4, y) : Date(s + 22, 3, y);
246 }
