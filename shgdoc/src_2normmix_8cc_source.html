58 double Normal_mixture::invcdf(double p) const {
59  if (p <= 0.0 || p >= 1.0)
60  throw Error();
61  double a = -1e3, b = 1e3, x = 0.0;
62  if (p <= cdf(a) || p >= cdf(b))
63  throw Error();
64 
65  // Bisection.
66  while (b - a > 1e-10)
67  if (cdf(x = 0.5 * (a + b)) < p)
68  a = x;
69  else
70  b = x;

95 void Normal_mixture::moments() {
96  if (moments_calculated)
97  return;
98  double m1 = 0.0, m2 = 0.0, m3 = 0.0, m4 = 0.0;
99 
100  // Calculate the first four raw moments of the mixture.
101  for (size_t i = 0; i < n; i++) {
102  const double wi = w[i];
103  const double mu1 = mu[i];
104  const double mu2 = sqr(mu1);
105  const double s2 = sqr(sigma[i]);
106  m1 += wi * mu1;
107  m2 += wi * (mu2 + s2);
108  m3 += wi * (mu2 + 3.0 * s2) * mu1;
109  m4 += wi * (sqr(mu2) + 6.0 * mu2 * s2 + 3.0 * sqr(s2));
110  }
111  mean_ = m1;
112  const double sm = sqr(m1);
113  const double var = m2 - sm;
114  if (var <= 0.0)
115  throw Error();
116  sdev_ = sqrt(var);
117  skew_ = (m3 - 3.0 * m1 * m2 + 2.0 * m1 * sm) / sdev_ / var;
118  curt_ = (m4 - 4.0 * m1 * m3 + 6.0 * sm * m2 - 3.0 * sqr(sm)) /
119  var / var - 3.0;
120  moments_calculated = true;
121 }
