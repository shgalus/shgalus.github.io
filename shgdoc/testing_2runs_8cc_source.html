24 double newton(int n, int k) {
25  SHG_ASSERT(n >= 0 && k >= 0);
26  if (n < k)
27  return 0.0;
28  int nk = n - k;
29  if (k > nk)
30  std::swap(k, nk);
31  double c = 1.0;
32  for (int i = 1; i <= k; i++)
33  c *= static_cast<double>(nk + i) / static_cast<double>(i);
34  return c;
35 }

37 // Returns P(K = k | N1 = n1, N2 = n2).
38 double pdf(int n1, int n2, int k) {
39  if (k % 2 == 0) {
40  const int l = k / 2;
41  return 2.0 * newton(n1 - 1, l - 1) *
42  newton(n2 - 1, l - 1) / newton(n1 + n2, n1);
43  } else {
44  const int l = (k - 1) / 2;
45  const double s1 = newton(n1 - 1, l - 1) * newton(n2 - 1, l);
46  const double s2 = newton(n1 - 1, l) * newton(n2 - 1, l - 1);
47  const double s3 = newton(n1 + n2, n1);
48  return (s1 + s2) / s3;
49  }
50 }

52 // Returns P(K <= k | N1 = n1, N2 = n2).
53 double cdf(int n1, int n2, int k) {
54  double p = 0.0;
55  for (int i = 2; i <= k; i++)
56  p += pdf(n1, n2, i);
57  return p;
58 }
