90 /**
91  * Calculates quotient and remainder of two integers.
92  *
93  * For each pair of integer numbers \f$a\f$ and \f$b\f$, where \f$ b
94  * \neq 0\f$, there exists exactly one pair of integer numbers \f$q\f$
95  * and \f$r\f$ (quotient and remainder), for which \f[ a = qb + r \;
96  * \mbox{and} \; 0 \leq r < |b|. \f] (See \cite mostowski-stark-1975,
97  * page 173).
98  *
99  * \implementation
100  *
101  * The standard \cite ansi-cpp-2012 states in paragraph 5.6.4:
102  *
103  * > The binary / operator yields the quotient, and the binary %
104  * > operator yields the remainder from the division of the first
105  * > expression by the second. If the second operand of / or % is zero
106  * > the behavior is undefined. For integral operands the / operator
107  * > yields the algebraic quotient with any fractional part discarded
108  * > (this is often called truncation towards zero); if the quotient
109  * > a/b is representable in the type of the result, (a/b)*b + a\%b is
110  * > equal to a.
111  *
112  * We denote by \f$\oslash\f$ division in C++. It is defined by \f[a
113  * \oslash b = \mathrm{trunc}\left(a / b \right) = \left\{
114  * \begin{array}{ll} \lfloor a / b \rfloor & \mbox{if $a / b \geq 0$,}
115  * \\ \lceil a / b \rceil & \mbox{if $a / b \leq 0$.} \end{array}
116  * \right. \f] Using inequalities \f$x - 1 < \lfloor x \rfloor \leq
117  * x\f$ and \f$x \leq \lceil x \rceil < x + 1\f$ for \f$x = a / b\f$
118  * we get four cases: \f{align*}{ a / b \geq 0, \quad b > 0, \quad 0 &
119  * \leq a - b \times (a \oslash b) < b, \\ a / b \geq 0, \quad b < 0,
120  * \quad b & < a - b \times (a \oslash b) \leq 0, \\ a / b \leq 0,
121  * \quad b > 0, \quad -b & < a - b \times (a \oslash b) \leq 0, \\ a /
122  * b \leq 0, \quad b < 0, \quad 0 & \leq a - b \times (a \oslash b) <
123  * -b. \f} In the first and the last cases, that is for \f$a \geq
124  * 0\f$, \f$a \oslash b\f$ and \f$a \% b\f$ give the correct results.
125  * To get the correct results in the remaining cases, that is for \f$a
126  * < 0\f$, if \f$a \% b \neq 0\f$, we have to decrease \f$a \% b\f$ by
127  * \f$b\f$ and increase \f$a \oslash b\f$ by 1 in the second case and
128  * increase \f$a \% b\f$ by \f$b\f$ and decrease \f$a \oslash b\f$ by
129  * 1 in the third case. Let us note that for \f$a \geq 0\f$ the
130  * results of operators \f$\oslash\f$ and \% are correct. Let us also
131  * note that the second and the third case which requires correction
132  * may be detected by testing whether \f$ a \% b \f$ is less than 0 or
133  * not.
134  */
135 template <class T>
136 struct Integer_division {
137  /**
138  * The constructor accepts \f$a\f$ and \f$b\f$ and the created
139  * object contains \f$q\f$ and \f$r\f$ described in
140  * Integer_division.
141  *
142  * \exception std::invalid_argument if \f$b = 0\f$
143  */
144  Integer_division(T a, T b);
145  T q; /**< quotient */
146  T r; /**< remainder */
147  /**
148  * Returns quotient described in Integer_division.

161 template <class T>
162 Integer_division<T>::Integer_division(T a, T b)
163  : q(), r() {
164  if (b == 0)
165  throw std::invalid_argument(
166  "Integer_division::Integer_division");
167  auto d = std::div(a, b);
168  assert((a < 0 && d.rem != 0) == (d.rem < 0));
169  if (d.rem < 0) {
170  if (b < 0) {
171  q = d.quot + 1;
172  r = d.rem - b;
173  } else {
174  q = d.quot - 1;
175  r = d.rem + b;
176  }
177  } else {
178  q = d.quot;
179  r = d.rem;
180  }
181 }

330  * Binary searching in a sorted subvector using relational operators.
331  * Searches for a key in sorted v[l], ..., v[u - 1]. Returns true if
332  * the key is found, then v[i] == key. If the key is not found, the
333  * value of is unchanged. The function works with both std::vector and
334  * SHG::Vector.
335  *
336  * \internal
337  *
338  * Previous version of this function was like this:
339  *
340  * template<class T>
341  * bool vbsearch(const T& key, const std::vector<T>& v,
342  * typename std::vector<T>::size_type l,
343  * typename std::vector<T>::size_type u,
344  * typename std::vector<T>::size_type& i) {
345  * if (u-- == 0 || u < l)
346  * return false;

349  * i = (l + u) / 2;

354  * } else if (key > v[i]) {
355  * if (i >= u)

360  * }
361  * }

365  if (u < l)
366  return false;

368  const auto it = std::lower_bound(v.begin() + l, end, key);
369  if (it != end && !(key < *it)) {
370  i = std::distance(v.begin(), it);
371  return true;
372  }
373  return false;
374 }
