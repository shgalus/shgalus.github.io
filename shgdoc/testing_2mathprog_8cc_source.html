220 /*
221  * \sum_{i = 1}^n (alpha[i] * x[i] - beta[i])^2, n big, alpha[i],
222  * beta[i] >= 0 minimizes to zero at x[i] = beta[i] / alpha[i].
223  */
224 void simple(int m, int n) {
225  int st;
226  Vecdouble p, C, b, x, x0;
227  Matdouble A;
228  double f, f0;
229  int i, j, k;
230  Vecdouble alpha(n), beta(n);
231  init_arr(m, n, p, C, A, b, x, x0);
232 
233  for (i = 0; i < n; i++) {
234  alpha(i) = i + 1;
235  beta(i) = (i + 1) * (i + 1);
236  }
237  j = n;
238  k = 0;
239  for (i = 0; i < n; i++) {
240  p(i) = -2.0 * alpha(i) * beta(i);
241  x0(i) = beta(i) / alpha(i);
242  C(k) = alpha(i) * alpha(i);
243  k += j--;
244  }
245  for (i = 0; i < m; i++) {
246  for (j = 0; j < n; j++)
247  A(i, j) = 1;
248  f = 0.0;
249  for (j = 0; j < n; j++)
250  f += beta(j) / alpha(j);
251  b(i) = f;
252  }
253  f0 = 0.0;
254  for (i = 0; i < n; i++)
255  f0 -= beta(i) * beta(i);
256  // alpha and beta can be released here
257  st = wolfe(p, C, A, b, x, f);
258  result(st, n, x, x0, f, f0);
259 }

261 /*
262  * The test data are generated in the following way. For a positive
263  * definite symmetric matrix H and such matrix U, vector v and vector z
264  * >= 0 that Uz = v, the form (Ux - v)^(T)H(Ux - v) has a minimum 0 at x
265  * = z. Thus the form p^(T)x + x^(T)C x, where C = U^(T)HU, p =
266  * -2v^(T)HU, has a minimum -v^(T)Hv at the same point. As a set of
267  * linear equality restrictions, a subset of the first m equalities from
268  * Ux = v may be taken. If U is a square invertible matrix, the minimum
269  * is unique.
270  *
271  * Here we take a Hilbert matrix as H, that is H(i, j) = 1 / (i + j +
272  * 1), i, j = 0, 1, ..., n - 1, and an upper triangle matrix of the form
273  * u[i][j] = 0 for i > j, u[i][j] = w^(j - i) for i <= j, i, j = 0, 1,
274  * ..., n - 1, w = 1.01, as U. We also take z = (1 1 ... 1)^(T).
275  */
276 void complex(int m, int n) {
277  const double w = 1.01;
278  int st;
279  Vecdouble p, C, b, x, x0,
280  &v = x; // v will be kept in x
281  Matdouble A, CC; // auxiliary, analogous to C
282  double f, f0;
283  double s, d;
284  int i, j, k;
285  init_arr(m, n, p, C, A, b, x, x0);
286  CC.assign(n, n, 0.0);
287 
288  // The first m rows of A and b.
289  for (i = 0; i < m; i++) {
290  s = 0.0;
291  d = 1.0;
292  for (j = i; j < n; j++) {
293  A(i, j) = d;
294  s += d;
295  d *= w;
296  }
297  b(i) = s;
298  }
299 
300  // Calculate v = Uz = U * (1 1 ... 1).
301  for (i = 0; i < n; i++) {
302  s = 0.0;
303  d = 1.0;
304  for (j = i; j < n; j++) {
305  s += d;
306  d *= w;
307  }
308  v(i) = s;
309  }
310 
311  // C <-- HU
312  for (i = 0; i < n; i++)
313  for (j = 0; j < n; j++) {
314  s = 0.0;
315  d = 1.0;
316  for (k = j; k >= 0; k--) {
317  s += d / (i + k + 1);
318  d *= w;
319  }
320  CC(i, j) = s;
321  }
322 
323  // p <-- -2v^(T)C
324  for (j = 0; j < n; j++) {
325  s = 0.0;
326  for (i = 0; i < n; i++)
327  s -= v(i) * CC(i, j);
328  p(j) = 2.0 * s;
329  }
330 
331  // Calculate f = -v^(T)Hv.
332  f0 = 0.0;
333  for (i = 0; i < n; i++) {
334  s = 0.0;
335  for (j = i + 1; j < n; j++)
336  s += v(j) / (i + j + 1);
337  s = 2.0 * s + v(i) / (2.0 * i + 1);
338  f0 -= s * v(i);
339  }
340 
341  // C <-- H^(T)C
342  for (i = 1; i < n; i++)
343  for (j = 0; j < n; j++)
344  CC(i, j) += CC(i - 1, j) * w;
345  for (i = 0; i < n; i++)
346  x0(i) = 1.0;
347 
348  k = 0;
349  for (i = 0; i < n; i++)
350  for (j = i; j < n; j++)
351  C(k++) = CC(i, j);
352 
353  st = wolfe(p, C, A, b, x, f);
354  result(st, n, x, x0, f, f0);
355 }
