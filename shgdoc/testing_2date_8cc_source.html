24 /*
25  * Returns day and month of Easter in Gregorian year y. This function is
26  * written exactly on the algorithm given in Donald E. Knuth, Sztuka
27  * programowania, t. I, Algorytmy podstawowe, p. 165-166, 538-540.
28  */
29 void knuth(int y, int* day, int* month) {
30  int g = y % 19 + 1;
31  int c = y / 100 + 1;
32  int x = (3 * c) / 4 - 12;
33  int z = (8 * c + 5) / 25 - 5;
34  int d = (5 * y) / 4 - x - 10;
35  int e = (11 * g + 20 + z - x) % 30;
36  if (y <= 1582) {
37  *day = *month = -1;
38  return;
39  }
40  if (e < 0)
41  e += 30;
42  if ((e == 25 && g > 11) || e == 24)
43  e++;
44  int n = 44 - e;
45  if (n < 21)
46  n += 30;
47  n = n + 7 - ((d + n) % 7);
48  if (n > 31) {
49  *day = n - 31;
50  *month = 4;
51  } else {
52  *day = n;
53  *month = 3;
54  }
55 }
