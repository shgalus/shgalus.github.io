100 unsigned long RNG::logarithmic(double p) {
101  SHG_VALIDATE(p > 0 && p < 1.0);
102  const double r = log(1.0 - p);
103  const double v = unipos();
104  if (v >= p)
105  return 1;
106  const double u = unipos();
107  const double q = 1.0 - exp(r * u);
108  if (v <= q * q) {
109  const double x = 1.0 + log(v) / log(q);

111  throw overflow_error("overflow in RNG::logarithmic");
112  return static_cast<unsigned long>(x);
113  }
114  if (v <= q)
115  return 2;
116  return 1;
117 }

155  SHG_VALIDATE(p >= 0.0 && p <= 1.0);
156  int a, b, k = 0;
157  double x;
158

163  if (x >= p) {

173  if (operator()() < p)

178 unsigned int RNG::poisson(double mu) {
179  SHG_VALIDATE(mu > 0.0);
180  unsigned int m, k = 0;
181  double x;
182 
183  while (mu > 10.0) {
184  m = static_cast<int>(0.875 * mu);
185  x = gamma_int(m);
186  if (x < mu) {
187  k += m;
188  mu -= x;
189  } else {
190  return k + binomial(mu / x, m - 1);
191  }
192  }
193  const double emu = exp(-mu);
194  double prod = 1.0;
195  do {
196  prod *= operator()();
197  k++;
198  } while (prod > emu);
199  return k - 1;
200 }

202 unsigned int RNG::negative_binomial(double t, double p) {
203  SHG_VALIDATE(t > 0.0 && p > 0.0 && p < 1.0);
204  const double mu = gamma(t) * (1.0 - p) / p;
205 
206  /*
207  * If the statement below were return (mu > 0.0) ? poisson(mu) :
208  * 0; there could be an assertion error in poisson(). Probably
209  * optimizing compiler does not store calculated mu in memory and
210  * compares mu as denormal number with 0. Under GCC it is enough
211  * to write if (mu < numeric_limits<double>::min()) cout << mu to
212  * see a lot of positive numbers less than min().
213  *
214  * Another problem is if it is justified to return 0 mu == 0. If
215  * mu == 0, however, the Poisson distribution can be considered
216  * as a degenerated distribution (probability of zero occurences
217  * of impossible event).
218  */
219 
220  return mu < numeric_limits<double>::min() ? 0 : poisson(mu);
221 }

247 double RNG::gamma_large(const double a) {
248  using Constants::pi;
249  const double sqa = sqrt(2.0 * a - 1.0);
250  double x, y, v;
251  do {
252  do {
253  y = tan(pi<double>() * operator()());
254  x = sqa * y + a - 1.0;
255  } while (x <= 0.0);
256  v = operator()();
257  } while (v > (1.0 + y * y) *
258  exp((a - 1.0) * log(x / (a - 1.0)) - sqa * y));
259  return x;

262 double RNG::gamma_frac(const double a) {
263  using Constants::e;
264  if (a == 0.0)
265  return 0.0;
266  const double p = e<double>() / (a + e<double>());
267  double q, x, u, v;
268  do {
269  u = operator()();
270  v = unipos();
271  if (u < p) {
272  x = exp((1.0 / a) * log(v));
273  q = exp(-x);
274  } else {
275  x = 1.0 - log(v);
276  q = exp((a - 1.0) * log(x));
277  }
278  } while (operator()() >= q);
279  return x;
280 }
