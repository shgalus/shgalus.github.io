185 /**
186  * Returns common logarithm of the relative error of approximation q
187  * of the correct value c. If c == 0.0, returns common logarithm of
188  * the absolute error.
189  *
190  * See \cite mccullough-1998 and \cite mccullough-1999, which at the
191  * time of writing could be found at
192  * http://www.pages.drexel.edu/~bdm25/publications.html.
193  */
194 double log_error(double q, double c) {
195  if (std::isnan(q) || std::isnan(c))
196  return 0.0;
197  if (q == c)
198  return numeric_limits<double>::digits10;
199  if (std::isinf(c) || std::isinf(q))
200  return 0.0;
201  const double le = c == 0.0 ?
202  -std::log10(std::abs(q)) :
203  -std::log10(std::abs(q - c) / std::abs(c));
204  SHG_ASSERT(!std::isnan(le));
205  return le;
206 }

208 void test_nist(ofstream& f, StRDLR& d, size_t n) {
209  double minle;
210  int mincsd;
211  auto update = [&minle, &mincsd](double q, double c) {
212  const double le = log_error(q, c);
213  int nd; // number of correct significant digits
214  if (le < 1.0)
215  nd = 0;
216  else if (le > numeric_limits<double>::digits10)
217  nd = numeric_limits<double>::digits10;
218  else
219  nd = static_cast<int>(le);
220  if (le < minle)
221  minle = le;
222  if (nd < mincsd)
223  mincsd = nd;
224  };
225 
226  d.active(n);
227  try {
228  const bool intercept = n != d.noint1 && n != d.noint2;
229  const int p = intercept ? d.k() + 1 : d.k();
230  OLS ols(d.X(), d.y(), intercept);
231  minle = numeric_limits<double>::max();
232  mincsd = numeric_limits<int>::max();
233 
234  // Compare values calculated by OLS with certified values.
235  for (int i = 0; i < p; i++) {
236  update(ols.beta()(i), d.b()(i));
237  update(ols.stderr()(i), d.sdevb()(i));
238  }
239  update(ols.ser(), d.sdevres());
240  update(ols.r2(), d.r2());
241  SHG_ASSERT(d.k() == static_cast<size_t>(d.dfreg()));
242  update(ols.ess(), d.ssreg());
243  update(ols.ess() / d.k(), d.msreg());
244  update(ols.fstat(), d.fstat());
245  SHG_ASSERT(ols.dof() == d.dfres());
246  update(ols.rss(), d.ssres());
247  update(ols.rss() / ols.dof(), d.msres());
248 
249  f << std::left << std::setw(10) << d.name() + string(": ")
250  << std::right << std::fixed << std::setprecision(4)
251  << "results correct to at least " << std::setw(2)
252  << mincsd << " digits (LRE = " << minle << ")\n";
253  if (n == d.wampler1 || n == d.wampler2) {
254  f << std::scientific << std::setprecision(12)
255  << "rss = " << ols.rss()
256  << ", orig rss = " << d.ssres()
257  << "\nfstat = " << ols.fstat()
258  << ", orig fstat = " << d.fstat() << '\n';
259  }
260  } catch (OLS::Singular_covariance_matrix) {
261  f << std::left << std::setw(10) << d.name() + string(": ")
262  << std::right << "could not invert matrix" << '\n';
263  }
264 }
