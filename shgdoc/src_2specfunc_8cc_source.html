19  using std::log;

22  static const double a2 = 1.0 / 1680.0;
23  static const double a3 = 1.0 / 1260.0;
24  static const double a4 = 1.0 / 360.0;
25  static const double a5 = 1.0 / 12.0;
26 
27  if (x <= 0.0)
28  throw std::invalid_argument(__func__);
29  double f = 0.0, z;
30  if (x < 7.0) {
31  f = x++;
32  while (x < 7.0)
33  f *= x++;

41 double alnorm(double x, bool upper) {
42  using std::numeric_limits;
43  using std::log;
44  using std::sqrt;
45 
46  /*
47  * Two machine dependent constants.
48  *
49  * The constant ltone should be set to the value at which the lower
50  * tail area becomes 1.0 to the accuracy of the machine. ltone = (n
51  * + 9 ) / 3 gives the required accuracy well enough for a machine
52  * producing n decimal digits. The constant utzero should be set to
53  * the value at which the upper tail area becomes 0.0 to the
54  * accuracy of the machine. This may be taken as sqrt(-2 * (ln(x) +
55  * 1)) - 0.3, where x is the smallest allowable real number.
56  */
57  static const double ltone =
58  (numeric_limits<double>::digits10 + 9) / 3.0;
59  static const double utzero =
60  sqrt(-2.0 * (log(numeric_limits<double>::min()) + 1.0)) - 0.3;
61

66  static const double a5 = 2.62433121679;

68  static const double a7 = 5.92885724438;

71  static const double b3 = 1.00000615302;
72  static const double b4 = 3.98064794e-4;

74  static const double b6 = 0.151679116635;
75  static const double b7 = 5.29330324926;
76  static const double b8 = 4.8385912808;
77  static const double b9 = 15.1508972451;
78  static const double b10 = 0.742380924027;
79  static const double b11 = 30.789933034;
80  static const double b12 = 3.99019417011;
81 
82  double z = 0.0;
83 
84  if (x < 0.0) {
85  upper = !upper;
86  x = -x;
87  }
88  if ((x <= ltone) || (upper && (x <= utzero))) {
89  const double y = 0.5 * x * x;
90  if (x > 1.28)
91  z = b1 * exp(-y) /
92  (x - b2 + b3 /
93  (x + b4 + b5 /
94  (x - b6 + b7 /
95  (x + b8 - b9 /
96  (x + b10 + b11 / (x + b12))))));
97  else
98  z = 0.5 - x *
99  (a1 - a2 * y /
100  (y + a3 - a4 /
101  (y + a5 + a6 / (y + a7))));
102  }
103  return upper ? z : 1.0 - z;
104 }

156 double gammad(double x, double p) {
157  using std::exp;
158  using std::abs;
159  using std::lgamma;
160  using std::log;
161  using std::numeric_limits;
162  using std::sqrt;
163 
164  static const double elimit = log(numeric_limits<double>::min());
165  static const double oflo = sqrt(numeric_limits<double>::max());
166  static const double plimit = 1000.0, xbig = 1e6, tol = 1e-8;
167 
168  if (x < 0.0 || p <= 0.0)
169  throw std::invalid_argument(__func__);
170  bool fin;
171  double pn1, pn2, pn3, pn4, pn5, pn6,
172  a, b, c, arg, an, rn, min, gam;
173 
174  if (x == 0.0)
175  return 0.0;
176  if (p > plimit) {
177  pn1 = exp(log(x / p) / 3.0) + 1.0 / (9.0 * p) - 1.0;
178  pn1 *= 3.0 * sqrt(p);
179  return normal_integral(pn1);
180  }
181  if (x > xbig)
182  return 1.0;
183  if (x <= 1.0 || x < p) {
184  arg = p * log(x) - x - lgamma(p + 1.0);
185  c = gam = 1.0;
186  a = p;
187  do {
188  gam += (c *= x / ++a);
189  } while (c > tol);
190  if ((arg += log(gam)) >= elimit)
191  return exp(arg);
192  else
193  return 0.0;
194  } else {
195  arg = p * log(x) - x - lgamma(p);
196  b = (a = 1.0 - p) + x + 1.0;
197  c = 0.0;
198  gam = (pn3 = (pn2 = x) + (pn1 = 1.0)) / (pn4 = x * b);
199  fin = false;
200  do {
201  an = ++a * ++c;
202  b += 2.0;
203  pn5 = b * pn3 - an * pn1;
204  pn6 = b * pn4 - an * pn2;
205  if (abs(pn6) > 0.0) {
206  if (tol < (min = tol * (rn = pn5 / pn6)))
207  min = tol;
208  if (abs(gam - rn) <= min)
209  fin = true;
210  else
211  gam = rn;
212  }
213  if (!fin) {
214  pn1 = pn3;
215  pn2 = pn4;
216  pn3 = pn5;
217  pn4 = pn6;
218  if (abs(pn5) >= oflo) {
219  pn1 /= oflo;
220  pn2 /= oflo;
221  pn3 /= oflo;
222  pn4 /= oflo;
223  }
224  }
225  } while (!fin);
226  if ((arg += log(gam)) >= elimit)
227  return 1.0 - exp(arg);
228  else
229  return 1.0;
230  }

243  const int m2 = df - 2, oe = df % 2;
244  f = 1.0;

246  f += (c *= b * (k - 1) / k);

255 double betain(const double x, const double p, const double q,
256  const double beta) {
257  using std::abs;
258  using std::exp;
259  using std::log;
260 
261  static const double accuracy = 1e-8;
262 
263  bool index;
264  int ns;
265  double bti = 1.0, ai = 1.0, cx = 1.0 - x, pp, psq = p + q, qq, rx,
266  temp, term = 1.0, xx;
267  if (p <= 0.0 || q <= 0.0 || x < 0.0 || x > 1.0)
268  throw std::invalid_argument(__func__);
269  if (x == 0.0 || x == 1.0)
270  return x;
271  if (p < psq * x) {
272  xx = cx; cx = x; pp = q; qq = p; index = true;
273  } else {
274  xx = x; pp = p; qq = q; index = false;
275  }
276  ns = static_cast<int>(qq + cx * psq);
277  rx = xx / cx;
278  temp = qq - ai;
279  if (ns == 0)
280  rx = xx;
281  for (;;) {
282  bti += (term *= temp * rx / (pp + ai));
283  temp = abs(term);
284  if (temp <= accuracy && temp <= accuracy * bti) {
285  bti *= exp(pp * log(xx) + (qq - 1.0) * log(cx) - beta) /
286  pp;
287  return index ? 1.0 - bti : bti;
288  }
289  ai++;
290  if (--ns < 0) {
291  temp = psq++;
292  } else {
293  temp = qq - ai;
294  if (ns == 0) rx = xx;
295  }
296  }

314 float digamma(float x) {
315  if (x <= 0.0)
316  throw std::invalid_argument(__func__);
317  if (x <= 1e-5)
318  return -0.5772156649f - 1.0f / x;
319  float dig = 0.0;
320  while (x < 8.5f)
321  dig -= 1.0f / x++;
322  /* Use Stirling's (actually de Moivre's) expansion if x > 8.5. */
323  float r = 1.0f / x;
324  dig += std::log(x) - 0.5f * r;
325  r *= r;
326  // The constants below are the third, fourth and fifth Stirling
327  // coefficients.
328  dig -= r * (8.333333333E-02f - r *
329  (8.3333333333E-03f - r * 3.968253968E-03f));
330  return dig;
331 }
