360 // Check if combinations are lexicographically ordered and if the
361 // number of combinations agrees with Newton formula.
362 void test(int n, int k) {
363  const size_t kk = k;
364  SHG::Comblex c(n, k);
365  vector<int> curr, prev;
366  int nr = 0;
367  do {
368  curr = c.get();
369  SHG_ASSERT(curr.size() == kk);
370  for (size_t i = 1; i < kk; i++)
371  SHG_ASSERT(curr[i - 1] < curr[i]);
372  if (nr > 0) {
373  size_t i = 0;
374  while (i < kk && prev[i] == curr[i])
375  i++;
376  SHG_ASSERT(i < kk);
377  SHG_ASSERT(prev[i] < curr[i]);
378  }
379  prev = curr;
380  nr++;
381  } while (c.next());
382  const long tnr = fact(n) / fact(k) / fact(n - k);
383  SHG_ASSERT(nr == tnr);
384 }
