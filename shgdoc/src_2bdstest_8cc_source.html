31  // The constant maxn_ is set in order to have n * (n - 1) / 2 < n
32  // * n / 2 <= std::numeric_limits<vdst>::max().
33  if (n < 1 || n > maxn_ || maxm < 2 ||
34  static_cast<vdst>(maxm) >= n || eps.size() < 1)
35  throw invalid_argument(__func__);
36  const double nd = n;
37  vector<bool> p(n * (n - 1) / 2);
38 
39  const auto sqr = [](double x) {return x * x;};
40 
41  // Returns true if |u[i] - u[j]| < eps.
42  const auto chi = [&p](vdst i, vdst j) -> bool {
43  if (i < j)
44  return p[(j - 1) * j / 2 + i];
45  else if (i > j)
46  return p[(i - 1) * i / 2 + j];
47  else
48  return true;
49  };

51  res_.resize(eps.size());
52 
53  for (vdst ieps = 0; ieps < eps.size(); ieps++) {
54  vdst count;
55  double tcount = 0.0;
56  // Calculate the matrix of distances and C_{1, n}(\epsilon)
57  // by \ref\label{eq:Cmn}.
58  {
59  const double epsi = eps[ieps];
60  vector<bool>::size_type k = 0;
61  count = 0;
62  for (vdst i = 1; i < n; i++)
63  for (vdst j = 0; j < i; j++)
64  if ((p[k++] = abs(u[i] - u[j]) < epsi))
65  count++;
66  }
67  const double c1 = 2.0 * count / n / (n - 1);
68 
69  // Calculate K by the formula \ref{eq:bds2.13s}.
70  for (vdst i = 0; i < n; i++) {
71  count = 0;
72  for (vdst j = 0; j < n; j++)
73  if (chi(i, j))
74  count++;
75  tcount += sqr(count);
76  }
77  const double k = tcount / (nd * nd * nd);
78 
79  // Calculate C defined by \ref{eq:bds2.12}. A simple
80  // relationship with C_{1, n}(\epsilon) is used.
81  const double c = (n - 1) * c1 / nd;
82 
83  res_[ieps].resize(maxm + 1);
84 
85  for (int m = 2; m <= maxm; m++) {
86  // Calculate C_{m, n} by \ref\label{eq:Cmn}.
87  const vdst nm1 = n - m + 1;
88  count = 0;
89  {
90  vdst i;
91  for (vdst s = 0; s < nm1; s++)
92  for (vdst t = s + 1; t < nm1; t++) {
93  for (i = 0;
94  i < static_cast<vdst>(m) &&
95  chi(s + i, t + i);
96  i++) {/* VOID */}
97  if (i == static_cast<vdst>(m))
98  count++;
99  }
100  }
101  const double cm = 2.0 * count / nm1 / (nm1 - 1);
102 
103  // Calculate V_{m, n} / \sqrt{n} by \ref{eq:simpleV}.
104  double v = 0.0;
105  for (int j = 1; j < m; j++)
106  v += pow(c, 2 * j) * pow(k, m - j);
107  v *= 2.0;
108  v += pow(k, m) + sqr((m - 1) * pow(c, m)) -
109  sqr(m) * k * sqr(pow(c, m - 1));
110  v *= 4.0 / nd;
111  v = sqrt(v);
112  double w = (cm - pow(c1, m)) / v;
113  if (!isfinite(w))
114  throw range_error(__func__);
115  res_[ieps][m].stat = w;
116  w *= isqrt2;
117  res_[ieps][m].pval =
118  w < 0.0 ?
119  0.5 + 0.5 * erf(w) :
120  0.5 * erfc(w);
121  }
122  }
