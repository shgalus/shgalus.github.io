24  * Natural logarithm of gamma function.
25  *
26  * Returns natural logarithm of \f$\Gamma(x)\f$ for \f$x > 0\f$.
27  * Accuracy is at least 8 significant decimal digits except the result
28  * is less than 0.1 when the accuracy is to at least 8 digits past
29  * decimal point. The gamma function is defined by
30  *
31  * \f$\Gamma(x) = \int_{0}^{+\infty} t^{x - 1}e^{-t}dt\f$ for \f$ x >
32  * 0\f$.
33  *
34  * \exception std::invalid_argument if \f$x \leq 0\f$
35  *
36  * Source: M. C. Pike, I. D. Hill, Algorithm ACM 291, in:
37  * \cite griffiths-hill-1986, p. 243-246.
38  *
39  * \deprecated std::lgamma() should be used since C++11.

93 /**
94  * Incomplete gamma integral.
95  *
96  * Returns \f$\frac{1}{\Gamma(p)} \int_0^x t^{p - 1} e^{-t} dt\f$
97  * for \f$p > 0\f$ and \f$x \geq 0\f$.
98  *
99  * \exception std::invalid_argument if \f$x < 0\f$ or \f$p \leq 0\f$
100  *
101  * \note If a random variable \f$X\f$ has a gamma distribution with
102  * probability density function \f[ f(x; b, p) = \left\{ \begin{array}
103  * {ll} \frac{b^p}{\Gamma(p)} x^{p - 1} e^{-bx} & \mbox{for $x > 0$} \\
104  * 0 & \mbox{for $x \leq 0$} \end{array} \right.\f] where \f$b\f$ and
105  * \f$p\f$ are positive parameters, then \f$P(X < x) =
106  * \mathit{gammad}(bx, p)\f$. If \f$X\f$ has a chi-squared distribution
107  * with \f$n\f$ degrees of freedom (\f$p = n / 2\f$, \f$b = 1 / 2\f$),
108  * then \f$P(X < x) = \mathit{gammad}(x / 2, n / 2)\f$.
109  *
110  * Source: \cite shea-1988.
111  */
112 double gammad(double x, double p);

169 /**
170  * The cumulative distribution function of the \f$F\f$ distribution.
171  *
172  * The Fisher-Snedecor \f$F\f$ distribution with \f$m > 0\f$ degrees
173  * of freedom in numerator and \f$n > 0\f$ degrees of freedom in
174  * denominator is a distribution of a random variable \f$\frac{Y/ m}{Z
175  * / n}\f$, where \f$Y\f$ and \f$Z\f$ are independent random variables
176  * with \f$\chi^2(m)\f$ and \f$\chi^2(n)\f$ distributions,
177  * respectively. The cumulative distribution function of this
178  * distribution is \cite bartoszewicz-1989, p. 43: \f[ F(x; m, n) =
179  * \frac{m^{m / 2}n^{n / 2} \Gamma((m + n) / 2)} {\Gamma(m / 2)
180  * \Gamma(n / 2)} \int_0^x \frac{t^{m / 2 - 1}}{(mt + n)^{(m + n) /
181  * 2}} dt \f] for \f$x > 0\f$ and \f$F(x; m, n) = 0\f$ for \f$x \leq
182  * 0\f$.
183  *
184  * For given \f$m \geq 1\f$, \f$n \geq 1\f$ and \f$x \in \mathbf{R}\f$,
185  * the function returns \f$F(x; m, n)\f$.
186  *
187  * \exception std::invalid_argument if \f$m < 1\f$ or \f$n < 1\f$
188  */
189 double cdffdist(int m, int n, double x);
