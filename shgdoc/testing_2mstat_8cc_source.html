24 // Cumulative distribution function of the Kolmogorov-Smirnov
25 // distribution as a sum of the series 1 - 2 \sum{j = 1}^{\infty}
26 // (-1)^{j - 1} exp(-2j^2z^2).
27 double F(const double x) {
28  if (x <= 0.0)
29  return 0.0;
30  double s = 0.0;
31  bool neg = false;
32  for (int j = 1;; j++) {
33  const double y = j * x;
34  const double z = std::exp(-2.0 * y * y);
35  if (z < std::numeric_limits<double>::min())
36  break;
37  if (neg) {
38  s -= z;
39  neg = false;
40  } else {
41  s += z;
42  neg = true;
43  }
44  }
45  s = 1.0 - 2.0 * s;
46  if (s < 0.0)
47  return 0.0;
48  return s;
49 }

229 // Calculate cdf of negative binomial distribution summing probability
230 // function.
231 double cdf_nb(double x, double t, double p) {
232  const double q = 1.0 - p;
233  SHG_ASSERT(t > 0.0 && p > 0.0 && q > 0.0);
234  if (x < 0.0)
235  return 0.0;
236  const int n = SHG::ifloor(x);
237  double b = 1.0, s = 1.0, z = 1.0;
238  for (int i = 1; i <= n; i++) {
239  b *= (t + i - 1) / i;
240  z *= q;
241  s += b * z;
242  }
243  return std::pow(p, t) * s;
244 }
