23 /**
24  * Returns the number of inversions in a sequence. The elements
25  * \f$a_i\f$ and \f$a_j\f$ of the sequence \f$a_0, a_1, \ldots, a_{n -
26  * 1}\f$ make an inversion if \f$j > i\f$ and \f$a_j < a_i\f$. The
27  * algorithm is based on merge-sort algorithm and is \f$O(n \log
28  * n)\f$. See \cite cormen-etal-2009, p. 42, \cite knuth-2002c,
29  * section 5.1.1.
30  *
31  * \tparam T A type with defined operator <.
32  *
33  * \param[in,out] a The sequence \f$a_0, a_1, \ldots, a_{n - 1}\f$,
34  * ordered increasingly on exit.
35  *
36  * \param[out] b Work array used for sorting, must be of size at least
37  * n.
38  *
39  * \param[in] n The number of elements in a.
40  *
41  * \return The number of inversions. It is equal to 0 for \f$n < 2\f$.
42  */
43 template <class T>
44 std::size_t ninv(T* a, T* b, std::size_t n) {
45  std::size_t ninv = 0;
46  for (size_t k = 1; k < n; k *= 2) {
47  const size_t k2 = 2 * k;
48  for (size_t p = 0; p + k < n; p += k2) {
49  const size_t q = p + k - 1;
50  const size_t r = std::min(q + k, n - 1);
51  size_t p1 = p, p2 = q + 1, i = p;
52  while (p1 <= q && p2 <= r)
53  if (a[p1] > a[p2]) {
54  ninv += (q - p1 + 1);
55  b[i++] = a[p2++];
56  } else {
57  b[i++] = a[p1++];
58  }
59  while (p1 <= q)
60  b[i++] = a[p1++];
61  while (p2 <= r)
62  b[i++] = a[p2++];
63  for (i = p; i <= r; i++)
64  a[i] = b[i];
65  }
66  }
67  return ninv;
68 }

77  *
78  * \param[in] n The number of elements in a.
79  *

81  *
82  * \sa ninv(T*, T*, std::size_t)
83  */

86  std::unique_ptr<T[]> a1(new T[n]), b(new T[n]);
87  std::copy_n(a, n, a1.get());
88  return ninv(a1.get(), b.get(), n);
