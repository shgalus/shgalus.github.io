22 /**
23  * Wald-Wolfowitz runs test. The function returns \f$p_1 = \Pr(K \leq
24  * k \mid N_1 = n_1, N_2 = n_2)\f$, \f$p_2 = \Pr(K \geq k \mid N_1 =
25  * n_1, N_2 = n_2)\f$, where \f$K\f$, \f$N_1\f$, \f$N_2\f$ are random
26  * variables representing number of runs, number of elements of type I
27  * and number of elements of type II in a sample, respectively. If
28  * \f$\mathit{exact} = \mathit{true}\f$ and \f$n_1 \geq 1, n_2 \geq
29  * 1\f$, the asymptotic distribution is used, otherwise the exact
30  * distribution of the number of runs is used.
31  *
32  * \exception std::range_error if \f$\mathit{exact} = \mathit{true}\f$
33  * and exact calculation results in underflow
34  *
35  * \implementation
36  *
37  * The following exact formulae have been used \cite fisz-1969, p.
38  * 437, \cite wald-wolfowitz-1940, p. 150: \f[ \Pr(K = k \mid N_1 =
39  * n_1, N_2 = n_2) = 2 \binom{n_1 - 1}{l - 1} \binom{n_2 - 1}{l - 1} /
40  * \binom{n}{n_1} \f] for \f$k = 2l\f$ and \f{multline*} \Pr(K = k
41  * \mid N_1 = n_1, N_2 = n_2) = \\ \left[ \binom{n_1 - 1}{l - 1}
42  * \binom{n_2 - 1}{l} + \binom{n_1 - 1}{l} \binom{n_2 - 1}{l - 1}
43  * \right] / \binom{n}{n_1} \f} for \f$k = 2l + 1\f$, where \f$n = n_1
44  * + n_2\f$ and \f$\binom{n}{k} = 0\f$ for \f$n < k\f$.
45  *
46  * The possible numbers of runs in a sample are included between 2 and
47  * \f$2 \min(n_1, n_2) + 1\f$ if \f$n_1 \neq n_2\f$ and 2 and \f$n\f$
48  * if \f$n_1 = n_2\f$.
49  *
50  * The expected value and variance of \f$K\f$ are \f{align*} E(K \mid
51  * N_1 = n_1, N_2 = n_2) &= \frac{2n_1n_2 + n}{n}, \\ D^2(K \mid N_1 =
52  * n_1, N_2 = n_2) &= \frac{2n_1n_2(2n_1n_2 - n)}{(n - 1)n^2}. \f}
53  *
54  * For asymptotic distribution, Wolf and Wolfowitz result
55  * \cite wald-wolfowitz-1940, p. 151 has been used that as \f$n_1\f$,
56  * \f$n_2\f$ tend to infinity with \f$n_1 / n_2\f$ being constant,
57  * \f$K\f$ is asymptotically normally distributed with mean \f$E(K
58  * \mid N_1 = n_1, N_2 = n_2)\f$ and variance \f$D^2(K \mid N_1 = n_1,
59  * N_2 = n_2)\f$.
60  *
61  * \note At the time of writing, the original article
62  * \cite wald-wolfowitz-1940 was available at <a href =
63  * "http://projecteuclid.org/">Project Euclid</a>.
64  */
65 void runs(std::size_t n1, std::size_t n2, std::size_t k, bool exact,
66  double& p1, double& p2);

69  * Returns the number of runs in a sample.
70  *
71  * The function returns the number of elements less than \f$z\f$ in
72  * the vector \f$x\f$ \f$n_1\f$, the number of elements greater or
73  * equal to \f$z\f$ \f$n_2\f$ and the number of runs \f$k\f$. The
74  * operation \f$<\f$ must be defined for the type \f$T\f$.
75  */
76 template <class T>
77 void count_runs(
78  const SHG::Vector<T>& x,
79  T z,
80  std::size_t& n1,
81  std::size_t& n2,
82  std::size_t& k) {
83  const auto n = x.size();
84  n1 = n2 = k = 0;
85  if (n == 0)
86  return;
87  bool s = x[0] < z;
88  k = 1;
89  for (const auto a : x)
90  if (a < z) {
91  n1++;
92  if (!s) {
93  s = true;
94  k++;
95  }
96  } else if (s) {
97  s = false;
98  k++;
99  }
100  n2 = n - n1;
101 }
