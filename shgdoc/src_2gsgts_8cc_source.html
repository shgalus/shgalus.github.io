57 void GSGTS::realft(const std::vector<std::complex<double>>& z,
58  std::vector<double>& X) {
59  const size_t n = z.size() - 1;
60  const double w = SHG::Constants::pi<double>() / n;
61  const double c = 1.0 / sqrt(n);
62  const double z0half = 0.5 * z[0].real();
63  const double znhalf = 0.5 * z[n].real();
64  size_t j, k;
65  double m, p, s;
66  int sign = 1;
67  for (j = 0; j < X.size(); j++) {
68  m = j * w;
69  s = 0.0;
70  for (k = 1; k < n; k++) {
71  p = k * m;
72  s += z[k].real() * cos(p) - z[k].imag() * sin(p);
73  }
74  X[j] = c * (z0half + s + sign * znhalf);
75  sign = -sign;
76  }
