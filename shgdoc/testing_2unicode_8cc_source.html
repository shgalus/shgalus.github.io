19 /**
20  * Encodes a Unicode character u in UTF-8. The returned value is the
21  * number of bytes in UTF-8 representation, namely 1, 2, 3 or 4. If
22  * the returned value is 1, the character c0 is the representation, if
23  * the returned value is 2, characters c0 and c1 make the
24  * representation and so on. If the returned value is 0, and invalid
25  * character u was given, that is u >= 0x110000u.
26  */
27 int utf8_encode(std::uint32_t u,
28  char* c0, char* c1, char* c2, char* c3) {
29  if (u < 0x0080u) {
30  /*
31  * 1 byte output
32  *
33  * 0xxxxxxx will be translated into:
34  *
35  * +----------+
36  * | 0xxxxxxx |
37  * +----------+
38  * | c0 |
39  * +----------+
40  */
41  const unsigned char x = u & 0x7fu;
42  *c0 = x;
43  return 1;
44  } else if (u < 0x0800u) {
45  /*
46  * 2 bytes output
47  *
48  * 00000yyy xxxxxxxx will be translated into:
49  *
50  * +----------+----------+
51  * | 110yyyxx | 10xxxxxx |
52  * +----------+----------+
53  * | c0 | c1 |
54  * +----------+----------+
55  */
56  const unsigned char y = (u & 0xff00u) / 0x0100u;
57  const unsigned char x = u & 0xffu;
58  *c0 = static_cast<char>(0xc0u + (y << 2) + (x >> 6));
59  *c1 = 0x80u + (x & 0x3f);
60  return 2;
61  } else if (u < 0x010000u) {
62  /*
63  * 3 bytes output
64  *
65  * yyyyyyyy xxxxxxxx will be translated into:
66  *
67  * +----------+----------+----------+
68  * | 1110yyyy | 10yyyyxx | 10xxxxxx |
69  * +----------+----------+----------+
70  * | c0 | c1 | c2 |
71  * +----------+----------+----------+
72  */
73  const unsigned char y = (u & 0xff00u) / 0x0100u;
74  const unsigned char x = u & 0xffu;
75  *c0 = 0xe0u + (y >> 4);
76  *c1 = static_cast<char>(0x80u + ((y & 0x0f) << 2) + (x >> 6));
77  *c2 = 0x80u + (x & 0x3f);
78  return 3;
79  } else if (u < 0x110000u) {
80  /*
81  * 4 bytes output
82  *
83  * 000zzzzz yyyyyyyy xxxxxxxx will be translated into:
84  *
85  * +----------+----------+----------+----------+
86  * | 11110zzz | 10zzyyyy | 10yyyyxx | 10xxxxxx |
87  * +----------+----------+----------+----------+
88  * | c0 | c1 | c2 | c3 |
89  * +----------+----------+----------+----------+
90  */
91  const unsigned char z = (u & 0xff0000u) / 0x010000u;
92  const unsigned char y = (u & 0xff00u) / 0x0100u;
93  const unsigned char x = u & 0xffu;
94  *c0 = 0xf0u + (z >> 2);
95  *c1 = static_cast<char>(
96  0x80u + ((z & 0x03) << 4) + ((y & 0xff) >> 4));
97  *c2 = static_cast<char>(
98  0x80u + ((y & 0x0f) << 2) + ((x & 0xc0) >> 6));
99  *c3 = 0x80u + (x & 0x3f);
100  return 4;
101  } else {
102  return 0;
103  }
104 }

109 void test1() {
110  char c0, c1, c2, c3;
111  int s;
112 
113  // Test all ASCII characters - one byte.
114  for (int c = 0; c < 0x80; c++) {
115  s = utf8_encode(c, &c0, &c1, &c2, &c3);
116  SHG_ASSERT(s == 1 && c0 == c);
117  }
118  // Test U+00A2 - two bytes.
119  s = utf8_encode(0x00a2, &c0, &c1, &c2, &c3);
120  SHG_ASSERT(s == 2 && c0 == '\xc2' && c1 == '\xa2');
121  // Test U+20AC - three bytes.
122  s = utf8_encode(0x20ac, &c0, &c1, &c2, &c3);
123  SHG_ASSERT(s == 3 && c0 == '\xe2' && c1 == '\x82' && c2 == '\xac');
124  // Test U+024B62 - four bytes.
125  s = utf8_encode(0x024b62, &c0, &c1, &c2, &c3);
126  SHG_ASSERT(s == 4 && c0 == '\xf0' && c1 == '\xa4' &&
127  c2 == '\xad' && c3 == '\xa2');
