359 /**
360  * Maximum likelihood estimation of the parameter of the logarithmic
361  * series distribution. Logarithmic series distribution has the
362  * probability function \cite devroye-1986, p. 545 \f[\Pr(X = n) =
363  * \frac{a}{n}p^n, \quad n \in \mathbf{N}, \quad 0 < p < 1,\f] where
364  * \f$a = -1 / \ln(1 - p)\f$. Its mean and variance are, respectively,
365  * \f[ \frac{ap}{1 - p}, \quad \frac{ap(1 - ap)}{(1 - p)^2}.\f] The
366  * maximum likelihood estimator of \f$p\f$ depends only on the sample
367  * mean. Given this mean, the function returns the estimate of
368  * \f$p\f$. The sample mean must be greater than 1 (sample mean equal
369  * to 1 implies degenerate distribution and sample mean less than 1 is
370  * impossible).
371  *
372  * \param [in] mean sample mean
373  *
374  * \exception SHG::Invalid_argument unless mean > 1.0
375  *
376  * \implementation The logarithm of likelihood \f$L\f$ of the sample
377  * \f$x_1, \ldots, x_n\f$ is \f[ \ln L(p) = -n \ln(-\ln(1 - p)) -
378  * \ln(x_1 \ldots x_n) + n \bar{x} \ln p, \f] where \f$\bar{x} = (x_1
379  * + \ldots + x_n) / n\f$. To maximize it, the second term on the
380  * right hand side may be left out and the remaining terms may be
381  * divided by \f$n\f$. Thus, the maximized function is \f[ f(p) =
382  * \bar{x} \ln(p) - \ln(-\ln(1 - p)). \f] This functions has the
383  * following derivatives: \f{align*} f'(p) &= \frac{\bar{x}}{p} +
384  * \frac{1}{(1 - p) \ln(1 - p)} \\ f''(p) &= - \frac{\bar{x}}{p^2} +
385  * \frac{1 + \ln(1 - p)}{[(1 - p) \ln(1 - p)]^2} \f} The algorithm
386  * solves the equation \f$f'(p) = 0\f$ by the method of bisection.
387  */
388 double mle_lsd(double mean);

405 /**
406  * Maximum likelihood estimation of the parameters of the negative
407  * binomial distribution. A random variable \f$X\f$ has a negative
408  * binomial distribution if its probability function is of the form
409  * \cite devroye-1986, p. 543, \cite knuth-2002b, p. 149 \f[ \Pr(X =
410  * n) = \binom{t - 1 + n}{n} p^t (1 - p)^n, \quad n = 0, 1, 2, \ldots,
411  * \f] where \f$t > 0\f$ and \f$0 < p < 1\f$ are real parameters. Its
412  * mean and variance are, respectively, \f[ \frac{t(1 - p)}{p}, \quad
413  * \frac{t(1 - p)}{p^2}. \f] Given the sample, the function calculates
414  * the maximum likelihood estimates of \f$t\f$ and \f$p\f$.
415  *
416  * \param [in] x vector of sample observations
417  * \param [out] t estimated parameter \f$t\f$
418  * \param [out] p estimated parameter \f$p\f$
419  *
420  * \exception std::invalid_argument if the sample size is less than 1
421  * or if a sample element is negative
422  *
423  * \exception std::runtime_error if the sample mean or variance is too
424  * close to 0
425  *
426  * \exception std::runtime_error if the point estimator of \f$p\f$
427  * turns out to be outside of \f$(0, 1)\f$ or if the estimator of
428  * \f$t\f$ turns out to be non-positive
429  *
430  * \implementation The logarithm of likelihood \f$L\f$ of the sample
431  * \f$x_1, \ldots, x_n\f$ is \f[ \ln L(t, p) = \sum_{i = 1}^n [\ln
432  * \Gamma(t + x_i) - \ln \Gamma(t)] - \sum_{i = 1}^n \ln x_i! + nt \ln
433  * p + n \bar{x} \ln(1 - p), \f] where \f$\bar{x} = (x_1 + \ldots +
434  * x_n) / n\f$. Solving \f[ \frac{\partial \ln L}{\partial p} =
435  * \frac{nt}{p} - \frac{n\bar{x}}{1 - p} = 0 \f] gives \f[ p =
436  * \frac{t}{t + \bar{x}}. \f] The equation \f[ \frac{\partial \ln
437  * L}{\partial t} = 0 \f] with \f$p\f$ substituted by \f[\frac{t}{t +
438  * \bar{x}}\f] takes the form \f[ f(t) \stackrel{\mathrm{def}}{=}
439  * \sum_{i = 1}^n \left[ \frac{\Gamma'(t + x_i)}{\Gamma(t + x_i)} -
440  * \frac{\Gamma'(t)}{\Gamma(t)} \right] + n \ln \frac{t}{t + \bar{x}}
441  * = 0. \f] Differentiating known formula \f[ \Gamma(t + x) = (t + x -
442  * 1)(t + x - 2) \ldots (t + 1) t \Gamma(t), \quad x \in \{1, 2,
443  * \ldots \}, \f] we get \f[ \frac{\Gamma'(t + x)}{\Gamma(t + x)} -
444  * \frac{\Gamma'(t)}{\Gamma(t)} = \frac{1}{t} + \frac{1}{t + 1} +
445  * \ldots \frac{1}{t + x - 2} + \frac{1}{t + x - 1}, \f] so \f[ f(t) =
446  * \sum_{\{i \colon x_i \neq 0 \}} \left[ \frac{1}{t} + \frac{1}{t +
447  * 1} + \ldots + \frac{1}{t + (x_i - 1)} \right] + n \ln \frac{t}{t +
448  * \bar{x}}. \f]
449  *
450  * The algorithm solves the equation \f$f(t) = 0\f$ by the Brent
451  * method \cite press-teukolsky-vetterling-flannery-2007, p. 454--456.
452  *
453  * It may be noted that \f[ f(t) = \sum_{i = 1}^n \psi(t + x_i) - n
454  * \psi(t) + n \ln \frac{t}{t + \bar{x}}, \f] where \f$\psi(t) =
455  * \Gamma'(t) / \Gamma(t)\f$ is the logarithmic derivative of
456  * \f$\Gamma\f$. The first and second derivatives of \f$f\f$ are easy
457  * to calculate, what might be useful implementing an algorithm with
458  * derivatives.
459  */
460 void mle_negative_binomial(const SHG::Vecint& x, double& t, double& p);
