35 void test_digamma() {
36  using SHG::Constants::gamma;
37  using SHG::digamma;
38  using std::log;
39  const double g = SHG::Constants::gamma<double>();
40  // digamma(1) = -gamma
41  SHG_ASSERT(abs(digamma(1) + gamma<double>()) < 5e-7);
42  // digamma(0.5) = -2 ln(2) - gamma
43  SHG_ASSERT(abs(digamma(0.5) + 2.0 * log(2.0) + gamma<double>())
44  < 9e-7);
45  // digamma(n) = -gamma + \sum_{k = 1}^{n - 1} (1/k) for n >= 2.
46  for (int n = 2; n <= 20; n++) {
47  double s = 0.0;
48  for (int k = n - 1; k > 0; k--)
49  s += 1.0 / k;
50  s -= g;
51  SHG_ASSERT(abs(digamma(n) - s) < 3e-7);
52  }
53  try {
54  digamma(0.0);
55  SHG_ASSERT(false);
56  } catch (std::invalid_argument) {}
57 }
