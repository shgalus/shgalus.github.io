19 /**
20  * Ordinary Least Squares. The class estimates a multiple regression
21  * model by the least squares method and provides summary statistics.
22  *
23  * Let \f$y = X\beta + u\f$ be a multiple linear regression model
24  * \cite maddala-2006, chapter 4. \f$y\f$ is an \f$n \times 1\f$
25  * vector of \f$n \geq 1\f$ observations on the dependent variable, X
26  * is an \f$n \times k\f$, \f$1 \leq k \leq n\f$, matrix of
27  * observations on the indepenent variables, \f$\beta\f$ is a \f$k
28  * \times 1\f$ vector of unknown parameters and \f$u\f$ is an \f$n
29  * \times 1\f$ vector of disturbances. The least squares estimator of
30  * \f$\beta\f$ is \f$\hat{\beta} = (X^TX)^{-1}X^Ty\f$.
31  *
32  * \todo Add methods for prediction.
33  *
34  * \ingroup mathematical_statistics
35  */

272  /**
273  * Calculates the Durbin-Watson \f$d\f$ statistic and its
274  * \f$p\f$-values for positive and negative autocorrelation. To
275  * check if the calculation succeeded, check dw_d(), dw_pvalpos()
276  * and dw_pvalneg().
277  *
278  * \implementation
279  *
280  * First it is checked if the intercept and at least one
281  * regressor are included. Then it is checked if the number of
282  * observations is at least 6 and (n_ - (k_ - 1)) >= 5 (these two
283  * conditions are present in tables \cite savin-white-1977). Then
284  * it is checked if rss is greater than 0. If any of these
285  * condition is not met, the function returns and OLS::dw_d_,
286  * OLS::dw_pvalpos_ and OLS::dw_pvalneg_ are not changed. Then
287  * the statistic d is calculated and assigned to OLS::dw_d_. Then
288  * p-value for positive autocorrelation is calculated and
289  * assigned to OLS::dw_pvalpos_; if the calculation fails,
290  * OLS::dw_pvalpos_ is not changed. Then p-value for negative
291  * autocorrelation is calculated and assigned to
292  * OLS::dw_pvalneg_; if the calculation fails, OLS::dw_pvalneg_
293  * is not changed.
294  *
295  * p-values are calculating as follows. It is clear that if
296  * random variables are such that \f$X \leq Y\f$, then \f$\Pr(Y <
297  * z) \leq \Pr(X < z)\f$. For the inequality \f$d_L \leq d \leq
298  * d_U\f$ this means that \f$\Pr(d_U < x) \leq \Pr(d < x) \leq
299  * \Pr(d_L < x)\f$.
300  *
301  * For testing \f$H_0 \colon \rho_1 = 0\f$ against \f$H_1 \colon
302  * \rho_1 > 0\f$, under \f$H_0\f$, low (near 0) values of \f$d\f$
303  * should reject \f$H_0\f$ as \f$\Pr(d < x) \leq \Pr(d_L < x) =
304  * \mathit{dwcdf}(x, \mathit{true})\f$. So, we accept
305  * \f$\mathit{dwcdf}(x, \mathit{true})\f$ as a p-value even
306  * though it is only an upper bound for the true p-value.
307  *
308  * For testing \f$H_0 \colon \rho_1 = 0\f$ against \f$H_1 \colon
309  * \rho_1 < 0\f$, under \f$H_0\f$, high (near 4) values of
310  * \f$d\f$ should reject \f$H_0\f$ as \f$\Pr(d \geq x) \leq
311  * \Pr(d_U \geq x) = 1 - \Pr(d_U < x) = 1 - \mathit{dwcdf}(x,
312  * \mathit{false})\f$. So, we accept \f$1 - \mathit{dwcdf}(x,
313  * \mathit{false})\f$ as a p-value even though it is only an
314  * upper bound for the true p-value.
315  *
316  * In both cases true p-values are lower then their upper bounds.
317  * On print, it should be marked that these are upper bounds, not
318  * p-values.
319  */
320  void dw();
