32 /**
33  * Compares the results of calls to dwcdf with those read from the
34  * file drbnwts1.txt. If the difference is too big, an SHG_ASSERT(0)
35  * is executed. The results in the input file come from the function
36  * dwcdf itself and were checked to be equal to the results of the
37  * function nag-prob-durbin-watson (g01epc) from the NAG C library,
38  * mark 7 (2001).
39  */
40 void cdftst() {
41  static const int n[20] = {
42  6, 7, 8, 9, 10, 15, 20, 25, 30, 35, 40,
43  45, 50, 60, 70, 80, 90, 100, 150, 200
44  };
45 
46  ifstream f(datapath("drbnwts1.txt").c_str());
47  SHG_ASSERT(bool(f));
48  {
49  string s;
50  getline(f, s);
51  SHG_ASSERT(bool(f));
52  }
53 
54  for (int i = 0; i < 20; i++) {
55  const int kk = min(10, n[i] - 5);
56  for (int k = 1; k <= kk; k++)
57  for (int j = 0; j <= 20; j++) {
58  const double x = j / 5.0;
59  const double
60  pdl = dwcdf(n[i], k, x, true, 1e-7, 17);
61  const double
62  pdu = dwcdf(n[i], k, x, false, 1e-7, 17);
63  double pdl0, pdu0;
64  f >> pdl0 >> pdu0;
65  SHG_ASSERT(bool(f));
66  SHG_ASSERT(abs(pdl - pdl0) < 5e-5);
67  SHG_ASSERT(abs(pdu - pdu0) < 5e-5);

72 /**
73  * The function compares the results of calls to ppdw with those read
74  * from the file drbnwts2.txt. This file contains a portion of the
75  * table from \cite savin-white-1977. If the difference is too big, an
76  * SHG_ASSERT(0) is executed.
77  *
78  * \warning In the table in drbnwts2.txt, for n = 6, 7, 8, 9, the
79  * inadequate positions were filled with 0.000 instead of dashes to
80  * keep one input format.
81  */
82 void ppdtst() {
83  double xl[5], xu[5], xl0, xu0, tol;
84  int n;
85 
86  ifstream f(datapath("drbnwts2.txt").c_str());
87  SHG_ASSERT(bool(f));

89  string s;

91  getline(f, s);
92  SHG_ASSERT(bool(f));

94 
95  for (int i = 0; i < 49; i++) {
96  f >> n
97  >> xl[0] >> xu[0]
98  >> xl[1] >> xu[1]
99  >> xl[2] >> xu[2]
100  >> xl[3] >> xu[3]
101  >> xl[4] >> xu[4];
102 
103  const int kk = min(5, n - 5);
104  for (int k = 1; k <= kk; k++) {
105  ppdw(n, k, 1e-2, &xl0, &xu0, 4.9e-5);
106  tol = 0.5e-3;
107  // These are differences between our function and the
108  // tables of Savin and White greater than 0.5e-3. There
109  // are 6 differences for d(l):
110  if (n == 10 && k == 1) tol = 0.6e-3;
111  if (n == 16 && k == 1) tol = 0.6e-3;
112  if (n == 18 && k == 2) tol = 0.6e-3;
113  if (n == 40 && k == 5) tol = 1.1e-3;
114  if (n == 200 && k == 2) tol = 0.8e-3;
115  if (n == 200 && k == 3) tol = 0.8e-3;
116  SHG_ASSERT(abs(xl0 - xl[k - 1]) < tol);
117  tol = 0.5e-3;
118  // and 9 differences for d(u):
119  if (n == 10 && k == 2) tol = 0.6e-3;
120  if (n == 11 && k == 4) tol = 0.6e-3;
121  if (n == 18 && k == 1) tol = 0.6e-3;
122  if (n == 20 && k == 2) tol = 0.6e-3;
123  if (n == 23 && k == 4) tol = 0.6e-3;
124  if (n == 29 && k == 4) tol = 0.6e-3;
125  if (n == 31 && k == 2) tol = 0.6e-3;
126  if (n == 200 && k == 2) tol = 0.8e-3;
127  if (n == 200 && k == 4) tol = 0.8e-3;
128  SHG_ASSERT(abs(xu0 - xu[k - 1]) < tol);
