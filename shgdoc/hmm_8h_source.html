84 /**
85  * Autocorrelation function of a hidden Markov model.
86  *
87  * Let \f$(X_t)_{t = 1}^{\infty}\f$ be an \f$m\f$-state hidden Markov
88  * model with transition matrix \f$\Gamma\f$ and stationary
89  * distribution \f$\delta\f$. For any function \f$g\f$ for which
90  * expectations exist \cite zucchini-macdonald-2009, p. 34,
91  * \f[
92  * E(g(X_t)) = \sum_{i = 1}^m \delta_i G_i = \delta G \mathbf{1}^T,
93  * \f]
94  * \f[
95  * E(g(X_t), g(X_{t + k})) = \sum_{i, j = 1}^m \delta_i G_i G_j
96  * \gamma_{ij}(k) = \delta G \Gamma^k G \mathbf{1}^T,
97  * \f]
98  * where \f$G_i = E(g(X_t) | C_t = i)\f$, \f$G = \mathrm{diag}(G_1,
99  * \ldots, G_m)\f$. Thus
100  * \f[
101  * \mathrm{cov}(g(X_t), g(X_{t + k})) = \delta G \Gamma^k G
102  * \mathbf{1}^T - (\delta G \mathbf{1}^T)^2
103  * \f]
104  * and
105  * \f[
106  * \mathrm{corr}(g(X_t), g(X_{t + k})) = \mathrm{cov}(g(X_t), g(X_{t +
107  * k})) / \mathrm{var}(g(X_t)).
108  * \f]
109  *
110  * This function, given \f$\delta\f$, \f$\Gamma\f$, \f$G\f$, \f$E\f$,
111  * \f$v\f$ and \f$K > 0\f$, calculates the autocorrelation function of
112  * \f$(X_t)\f$ for the lags \f$0, 1, \ldots, K\f$ and puts the
113  * autocorrelation coefficients in <tt>r[K + 1]</tt> so that \c r[k]
114  * contains autocorrelation coefficient \f$r_k\f$ for \f$k = 0,
115  * \ldots, K\f$. The vector \c r is resized in the function to get
116  * proper dimension.
117  *
118  * \param[in] delta stationary distribution
119  * \param[in] gamma transition matrix
120  * \param[in] G \f$E(g(X_t)\;|\;C_t = i)\f$
121  * \param[in] E \f$E(g(X_t))\f$
122  * \param[in] v \f$\mbox{var}(g(X_t))\f$
123  * \param[in] K maximum lag
124  * \param[out] r autocorrelations, r[0] == 1
125  */
126 void corr_hmm(const Vecdouble& delta,
127  const Matdouble& gamma,
128  const Vecdouble& G,
129  double E,
130  double v,
131  int K,
132  Vecdouble& r);

134 /**
135  * Autocorrelation function of a normal hidden Markov model.
136  *
137  * Calculates autocorrelation function of \f$g(X_t)\f$ for a
138  * stationary normal hidden Markov model. Three values of \c
139  * function_g may be given: 0 for \f$g(X) = X\f$, 1 for \f$g(X) =
140  * |X|\f$, 2 for \f$g(X) = X * X\f$.
141  *
142  * \implementation For \f$g(X) = X\f$ the implementation is
143  * straightforward: \f[ E(g(X_t) | C_t = i) = \mu_i, \f] \f[
144  * E(g(X_t))= \sum_{i = 1}^m \delta_i \mu_i, \f] \f[
145  * \mathrm{var}(g(X_t)) = \sum_{i = 1}^m \delta_i (\sigma_i^2 +
146  * \mu_i^2) - [E(g(X_t))]^2. \f]
147  * For \f$g(X) = |X|\f$, \f[ E(g(X_t) | C_t = i) = \sigma_i
148  * \sqrt{\frac{2}{\pi}} \exp \left( -\frac{\mu_i^2}{2
149  * \sigma_i^2}\right) + \mu_i[1 - 2 \Phi(-\mu_i / \sigma_i)], \f] \f[
150  * E(g(X_t)) = \sum_{i = 1}^m \delta_i E(g(X_t) | C_t = i), \f] \f[
151  * \mathrm{var}(g(X_t)) = E(X_t^2) - [E(g(X_t))]^2 = \sum_{i = 1}^m
152  * \delta_i (\mu_i^2 + \sigma_i^2) - [E(g(X_t))]^2, \f] where
153  * \f$\Phi\f$ is the cumulative distribution function of standard
154  * normal distribution.
155  * For \f$g(X) = X^2\f$ (cf. implementation of \c Normal_mixture), \f[
156  * E(g(X_t) | C_t = i) = \mu_i^2 + \sigma_i^2, \f] \f[ E(g(X_t)) =
157  * \sum_{i = 1}^m \delta_i (\mu_i^2 + \sigma_i^2), \f] \f[
158  * \mathrm{var}(g(X_t)) = E(X_t^4) - [E(g(X_t))]^2 = \sum_{i = 1}^m
159  * \delta_i (\mu_i^4 + 6 \mu_i^2 \sigma_i^2 + 3 \sigma_i^4) -
160  * [E(g(X_t))]^2. \f]
161  *
162  * \param[in] delta stationary distribution
163  * \param[in] gamma transition matrix
164  * \param[in] mu state-dependent means
165  * \param[in] sigma state-dependent std. dev.
166  * \param[in] function_g see above
167  * \param[in] K maximum lag
168  * \param[out] r autocorrelations, r[0] == 1
169  */
170 void corr_nhmm(const Vecdouble& delta,
171  const Matdouble& gamma,
172  const Vecdouble& mu,
173  const Vecdouble& sigma,
174  int function_g,
175  int K,
